\section{Failure Classification}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Failure Classification\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Database System Recovery}
    \begin{itemize}[<+->]
        \item All database reads/writes are within a transaction
        \item Transactions have the 'ACID' properties
        \begin{itemize}
            \item[$\triangleright$] Atomicity - all or nothing
            \item[$\triangleright$] Consistency - preserves database integrity
            \item[$\triangleright$] Isolation - execute as if they were run alone
            \item[$\triangleright$] Durability - results are not lost by a failure
        \end{itemize}
        \item Concurrency Control guarantees I, contributes to C
        \item Application program guarantees C
        \item Recovery subsystem guarantees A \& D, contributes to C
    \end{itemize}
\end{frame}

\begin{frame}{Failure Classification}
    \begin{itemize}[<+->]
        \item \textbf{Transaction failure:}
        \begin{itemize}
            \item[$\triangleright$] Logical errors : transaction cannot complete due to some internal error condition
            \item[$\triangleright$] System errors : the database system must terminate an active transaction due to an error condition (for example, deadlock)
        \end{itemize}
        \item \textbf{System crash:} a power failure or other hardware or software failure causes the system to crash
        \begin{itemize}
            \item[$\triangleright$] Fail-stop assumption : non-volatile storage contents are assumed to not be corrupted as result of a system crash
            \item[$\triangleright$] Database systems have numerous integrity checks to prevent corruption of disk data
        \end{itemize}
        \item \textbf{Disk failure:} a head crash or similar disk failure destroys all or part of disk storage
        \begin{itemize}
            \item[$\triangleright$] Destruction is assumed to be detectable
            \item[$\triangleright$] Disk drives use checksums to detect failures
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithms}
    \begin{itemize}[<+->]
        \item Consider transaction $T_i$ that transfers \$50 from account A to account B
        \begin{itemize}
            \item[$\triangleright$] Two updates: subtract 50 from A and add 50 to B
            \item[$\triangleright$] Transaction $T_i$ requires updates to A and B to be output to the database
        \end{itemize}
        \item A failure may occur after one of these modifications have been made but before both of them are made
        \item Modifying the database without ensuring that the transaction will commit may leave the database in an inconsistent state
        \item Not modifying the database may result in lost updates if failure occurs just after transaction commits
        \item Recovery algorithms have two parts
        \begin{itemize}
            \item[$\triangleright$] a) Actions taken during normal transaction processing to ensure enough information exists to recover from failures
            \item[$\triangleright$] b) Actions taken after a failure to recover the database contents to a state that ensures atomicity, consistency and durability
        \end{itemize}
    \end{itemize}
\end{frame}

\section{Storage Structure}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Storage Structure\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Storage Structure}
    \begin{itemize}[<+->]
        \item \textbf{Volatile Storage:}
        \begin{itemize}
            \item[$\triangleright$] does not survive system crashes
            \item[$\triangleright$] examples: main memory, cache memory
        \end{itemize}
        \item \textbf{Nonvolatile Storage:}
        \begin{itemize}
            \item[$\triangleright$] survives system crashes
            \item[$\triangleright$] examples: disk, tape, flash memory, non-volatile (battery backed up) RAM
            \item[$\triangleright$] but may still fail, losing data
        \end{itemize}
        \item \textbf{Stable Storage:}
        \begin{itemize}
            \item[$\triangleright$] a mythical form of storage that survives all failures
            \item[$\triangleright$] approximated by maintaining multiple copies on distinct non-volatile media
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Stable Storage Implementation}
    \begin{itemize}[<+->]
        \item Maintain multiple copies of each block on separate disks
        \begin{itemize}
            \item[$\triangleright$] copies can be at remote sites to protect against disasters such as fire or flooding
        \end{itemize}
        \item Failure during data transfer can still result in inconsistent copies. Block transfer can result in
        \begin{itemize}
            \item[$\triangleright$] Successful completion
            \item[$\triangleright$] Partial failure: destination block has incorrect information
            \item[$\triangleright$] Total failure: destination block was never updated
        \end{itemize}
        \item Protecting storage media from failure during data transfer (one solution):
        \begin{itemize}
            \item[$\triangleright$] Execute output operation as follows (assuming two copies of each block):
            \item[$\triangleright$] Write the information onto the 1st physical block
            \item[$\triangleright$] When the 1st write is successful, write the same information onto the 2nd physical block
            \item[$\triangleright$] The output is completed only after the second write successfully completes
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Stable Storage Implementation (2)}
    \begin{itemize}[<+->]
        \item Protecting storage media from failure during data transfer (cont.):
        \item Copies of a block may differ due to failure during output operation
        \item To recover from failure:
        \begin{itemize}
            \item[$\triangleright$] First find inconsistent blocks:
            \item[$\triangleright$] \textbf{Expensive solution:} Compare the two copies of every disk block
            \item[$\triangleright$] \textbf{Better solution:} 
            \item[$\bullet$] Record in-progress disk writes on non-volatile storage (Non-volatile RAM or special area of disk)
            \item[$\bullet$] Use this information during recovery to find blocks that may be inconsistent, and only compare copies of these
            \item[$\bullet$] Used in hardware RAID systems
        \end{itemize}
        \item If either copy of an inconsistent block is detected to have an error (bad checksum), overwrite it by the other copy
        \item If both have no error, but are different, overwrite the second block by the first block
    \end{itemize}
\end{frame}

\begin{frame}{Data Access}
    \begin{itemize}[<+->]
        \item Physical Blocks are those blocks residing on the disk
        \item System Buffer Blocks are the blocks residing temporarily in main memory
        \item Block movements between disk and main memory are initiated through the following two operations:
        \begin{itemize}
            \item[$\triangleright$] \texttt{input(B)} transfers the physical block B to main memory
            \item[$\triangleright$] \texttt{output(B)} transfers the buffer block B to the disk, and replaces the appropriate physical block there
        \end{itemize}
        \item We assume, for simplicity, that each data item fits in, and is stored inside, a single block
    \end{itemize}
\end{frame}

\begin{frame}{Data Access (2)}
    \begin{itemize}[<+->]
        \item Each transaction $T_i$ has its private work-area in which local copies of all data items accessed and updated by it are kept
        \item $T_i$'s local copy of a data item X is denoted by $x_i$
        \item $B_X$ denotes block containing X
        \item Transferring data items between system buffer blocks and its private work-area done by:
        \begin{itemize}
            \item[$\triangleright$] \texttt{read(X)} assigns the value of data item X to the local variable $x_i$
            \item[$\triangleright$] \texttt{write(X)} assigns the value of local variable $x_i$ to data item X in the buffer block
        \end{itemize}
        \item Transactions
        \begin{itemize}
            \item[$\triangleright$] Must perform \texttt{read(X)} before accessing X for the first time (subsequent reads can be from local copy)
            \item[$\triangleright$] The \texttt{write(X)} can be executed at any time before the transaction commits
            \item[$\triangleright$] Note that \texttt{output($B_X$)} need not immediately follow \texttt{write(X)}. System can perform the output operation when it deems fit
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery and Atomicity}
    \begin{itemize}[<+->]
        \item To ensure atomicity despite failures, we first output information describing the modifications to stable storage without modifying the database itself
        \item We study Log-based Recovery Mechanisms
        \begin{itemize}
            \item[$\triangleright$] We first present key concepts
            \item[$\triangleright$] And then present the actual recovery algorithm

        \item Less used alternative: Shadow Paging

\section{Log-Based Recovery}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Log-Based Recovery\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Log-Based Recovery}
    \begin{itemize}[<+->]
        \item A log is kept on stable storage
        \item The log is a sequence of log records, which maintains information about update activities on the database
        \item When transaction $T_i$ starts, it registers itself by writing a record \texttt{<$T_i$ start>} to the log
        \item Before $T_i$ executes \texttt{write(X)}, a log record \texttt{<$T_i$, X, $V_1$, $V_2$>} is written, where $V_1$ is the value of X before the write (old value), and $V_2$ is the value to be written to X (new value)
        \item When $T_i$ finishes its last statement, the log record \texttt{<$T_i$ commit>} is written
    \end{itemize}
\end{frame}

\begin{frame}{Database Modification Schemes}
    \begin{itemize}[<+->]
        \item The \textbf{immediate-modification} scheme allows updates of an uncommitted transaction to be made to the buffer, or the disk itself, before the transaction commits
        \begin{itemize}
            \item[$\triangleright$] Update log record must be written before a database item is written
            \item[$\triangleright$] We assume that the log record is output directly to stable storage
            \item[$\triangleright$] Output of updated blocks to disk storage can take place at any time before or after transaction commit
            \item[$\triangleright$] Order in which blocks are output can be different from the order in which they are written
        \end{itemize}
        \item The \textbf{deferred-modification} scheme performs updates to buffer/disk only at the time of transaction commit
        \begin{itemize}
            \item[$\triangleright$] Simplifies some aspects of recovery
            \item[$\triangleright$] But has overhead of storing local copy
        \end{itemize}
        \item We cover here only the immediate-modification scheme
    \end{itemize}
\end{frame}

\begin{frame}{Transaction Commit}
    \begin{itemize}[<+->]
        \item A transaction is said to have committed when its commit log record is output to stable storage
        \item All previous log records of the transaction must have been output already
        \item Writes performed by a transaction may still be in the buffer when the transaction commits, and may be output later
    \end{itemize}
\end{frame}

\end{frame}

\begin{frame}{Undo and Redo Operations}
    \begin{itemize}[<+->]
        \item \textbf{Undo} of a log record \texttt{<$T_i$, X, $V_1$, $V_2$>} writes the old value $V_1$ to X
        \item \textbf{Redo} of a log record \texttt{<$T_i$, X, $V_1$, $V_2$>} writes the new value $V_2$ to X
        \item \textbf{Undo and Redo of Transactions}
        \begin{itemize}
            \item[$\triangleright$] \texttt{undo($T_i$)} restores the value of all data items updated by $T_i$ to their old values, going backwards from the last log record for $T_i$
            \item[$\triangleright$] Each time a data item X is restored to its old value $V_1$, a special log record (called \textbf{redo-only}) \texttt{<$T_i$, X, $V_1$>} is written out
            \item[$\triangleright$] When undo of a transaction is complete, a log record \texttt{<$T_i$ abort>} is written out (to indicate that the undo was completed)
            \item[$\triangleright$] \texttt{redo($T_i$)} sets the value of all data items updated by $T_i$ to the new values, going forward from the first log record for $T_i$
            \item[$\triangleright$] No logging is done in this case
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Undo and Redo Operations (2)}
    \begin{itemize}[<+->]
        \item The undo and redo operations are used in several different circumstances:
        \begin{itemize}
            \item[$\triangleright$] The \texttt{undo} is used for transaction rollback during normal operation in case a transaction cannot complete its execution due to some logical error
            \item[$\triangleright$] The \texttt{undo} and \texttt{redo} operations are used during recovery from failure
        \end{itemize}
        \item We need to deal with the case where during recovery from failure another failure occurs prior to the system having fully recovered
    \end{itemize}
\end{frame}

\begin{frame}{Undo and Redo on Normal Transaction Rollback}
    \begin{itemize}[<+->]
        \item Let $T_i$ be the transaction to be rolled back
        \item Scan log backwards from the end, and for each log record of $T_i$ of the form \texttt{<$T_i$, $X_j$, $V_1$, $V_2$>}
        \begin{itemize}
            \item[$\triangleright$] Perform the undo by writing $V_1$ to $X_j$,
            \item[$\triangleright$] Write a log record \texttt{<$T_i$, $X_j$, $V_1$>}
            \item[$\triangleright$] such log records are called \textbf{Compensation Log Records}
        \end{itemize}
        \item Once the record \texttt{<$T_i$ start>} is found stop the scan and write the log record \texttt{<$T_i$ abort>}
    \end{itemize}
\end{frame}

\begin{frame}{Undo and Redo on Recovering from Failure}
    \begin{itemize}[<+->]
        \item When recovering after failure:
        \item Transaction $T_i$ needs to be undone if the log
        \begin{itemize}
            \item[$\triangleright$] contains the record \texttt{<$T_i$ start>},
            \item[$\triangleright$] but does not contain either the record \texttt{<$T_i$ commit>} or \texttt{<$T_i$ abort>}
        \end{itemize}
        \item Transaction $T_i$ needs to be redone if the log
        \begin{itemize}
            \item[$\triangleright$] contains the records \texttt{<$T_i$ start>}
            \item[$\triangleright$] and contains the record \texttt{<$T_i$ commit>} or \texttt{<$T_i$ abort>}
        \end{itemize}
        \item It may seem strange to redo transaction $T_i$ if the record \texttt{<$T_i$ abort>} record is in the log
        \begin{itemize}
            \item[$\triangleright$] To see why this works, note that if \texttt{<$T_i$ abort>} is in the log, so are the redo-only records written by the undo operation. Thus, the end result will be to undo $T_i$'s modifications in this case. This slight redundancy simplifies the recovery algorithm and enables faster overall recovery time
            \item[$\triangleright$] such a redo redoes all the original actions including the steps that restored old value - Known as \textbf{Repeating History}
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Checkpoints}
    \begin{itemize}[<+->]
        \item Redoing/undoing all transactions recorded in the log can be very slow
        \item Processing the entire log is time-consuming if the system has run for a long time
        \item We might unnecessarily redo transactions which have already output their updates to the database
        \item Streamline recovery procedure by periodically performing \textbf{checkpointing}
        \item All updates are stopped while doing checkpointing
        \begin{itemize}
            \item[$\triangleright$] a) Output all log records currently residing in main memory onto stable storage
            \item[$\triangleright$] b) Output all modified buffer blocks to the disk
            \item[$\triangleright$] c) Write a log record \texttt{<checkpoint L>} onto stable storage where L is a list of all transactions active at the time of checkpoint
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Checkpoints (2)}
    \begin{itemize}[<+->]
        \item During recovery we need to consider only the most recent transaction $T_i$ that started before the checkpoint, and transactions that started after $T_i$
        \item Scan backwards from end of log to find the most recent \texttt{<checkpoint L>} record
        \begin{itemize}
            \item[$\triangleright$] Only transactions that are in L or started after the checkpoint need to be redone or undone
            \item[$\triangleright$] Transactions that committed or aborted before the checkpoint already have all their updates output to stable storage
        \end{itemize}
        \item Some earlier part of the log may be needed for undo operations
        \begin{itemize}
            \item[$\triangleright$] Continue scanning backwards till a record \texttt{<$T_i$ start>} is found for every transaction $T_i$ in L
            \item[$\triangleright$] Parts of log prior to earliest \texttt{<$T_i$ start>} record above are not needed for recovery, and can be erased whenever desired
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Module Summary}
    \begin{itemize}[<+->]
        \item Failures may be due to variety of sources - each needs a strategy for handling
        \item A proper mix and management of volatile, non-volatile and stable storage can guarantee recovery from failures and ensure Atomicity, Consistency and Durability
        \item Log-based recovery is efficient and effective
    \end{itemize}
