\section{File Organization}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}File Organization\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{File Organization}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item A database is a collection of files. A file is a sequence of records. A record is a sequence of fields.
            \item \textbf{One approach:}
            \begin{itemize}[<+->]
                \item assume record size is fixed
                \item each file has records of one particular type only
                \item different files are used for different relations
            \end{itemize}
            \item This case is easiest to implement; will consider variable length records later
            \item A database file is partitioned into fixed-length storage units called blocks
            \item Blocks are units of both storage allocation and data transfer
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Fixed-Length Records}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{block}{}
                \begin{itemize}[<+->]
                    \item \textbf{Simple approach}: Store record $i$ starting from byte $n \times (i - 1)$, where $n$ is the size of each record.
                    \item Record access is simple but records may cross blocks.
                    \item \textbf{Modification}: do not allow records to cross block boundaries.
                    \item \textbf{Deletion of record $i$}: Alternatives:
                    \begin{itemize}[<+->]
                        \item move records $i + 1, \dots, n$ to $i, \dots, n - 1$
                        \item move record $n$ to $i$
                        \item do not move records, but link all free records on a free list
                    \end{itemize}
                \end{itemize}
            \end{block}
        \end{column}

\begin{frame}{Free Lists}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{block}{}
                \begin{itemize}[<+->]
                    \item Store the address of the first deleted record in the file header
                    \item Use this first record to store the address of the second deleted record, and so on
                    \item Consider these stored addresses as pointers since they point to the location of a record
                    \item More space efficient representation: reuse space for normal attributes of free records to store pointers (No pointers stored in in-use records)
                \end{itemize}
            \end{block}
        \end{column}

\section{Variable-Length Records}

\begin{frame}{Variable-Length Records}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Variable-length records arise in database systems in several ways:
            \begin{itemize}[<+->]
                \item Storage of multiple record types in a file
                \item Record types that allow variable lengths for one or more fields such as strings (\texttt{varchar})
                \item Record types that allow repeating fields (used in some older data models)
            \end{itemize}
            \item Attributes are stored in order
            \item Variable length attributes represented by fixed size (offset, length), with actual data stored after all fixed length attributes
            \item Null values represented by null-value bitmap
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Variable-Length Records (2)}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{block}{}
                \begin{itemize}[<+->]
                    \item \textbf{Slotted Page header} contains:
                    \begin{itemize}[<+->]
                        \item number of record entries
                        \item end of free space in the block
                        \item location and size of each record
                    \end{itemize}
                    \item Records can be moved around within a page to keep them contiguous with no empty space between them; entry in the header must be updated
                    \item Pointers should not point directly to record - instead they should point to the entry for the record in header
                \end{itemize}
            \end{block}
        \end{column}

\section{Organization of Records in Files}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Organization of Records in Files\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Organization of Records in Files}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Heap} : A record can be placed anywhere in the file where there is space
            \item \textbf{Sequential} : Store records in sequential order, based on the value of the search key of each record
            \item \textbf{Hashing} : A hash function computed on some attribute of each record; the result specifies in which block of the file the record should be placed
            \item Records of each relation may be stored in a separate file. In a \textbf{multitable clustering file organization} records of several different relations can be stored in the same file
            \item \textbf{Motivation}: store related records on the same block to minimize I/O
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Sequential File Organization}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{}
                \begin{itemize}[<+->]
                    \item Suitable for applications that require sequential processing of the entire file
                    \item The records in the file are ordered by a search-key
                \end{itemize}
            \end{block}

\begin{frame}{Sequential File Organization (2)}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{block}{}
                \begin{itemize}[<+->]
                    \item \textbf{Deletion}: Use pointer chains
                    \item \textbf{Insertion}: Locate the position where the record is to be inserted
                    \begin{itemize}[<+->]
                        \item if there is free space insert there
                        \item if no free space, insert the record in an overflow block
                    \end{itemize}
                    \item In either case, pointer chain must be updated
                    \item Need to reorganize the file from time to time to restore sequential order
                \end{itemize}
            \end{block}

\section{Data Dictionary Storage}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Data Dictionary Storage\par%
  \end{beamercolorbox}
  \vfill

\begin{frame}{Data Dictionary Storage}
    \begin{block}{}
        Data Dictionary (also, System Catalog) stores metadata (data about data) such as:
        \begin{itemize}
            \item \textbf{Information about relations}
            \begin{itemize}
                \item names of relations, names, types and lengths of attributes of each relation
                \item names and definitions of views, integrity constraints
            \end{itemize}
            \item \textbf{User and accounting information}, including passwords
            \item \textbf{Statistical and descriptive data} : number of tuples in each relation
            \item \textbf{Physical file organization information} : How relation is stored (sequential/hash...), Physical location of relation
            \item \textbf{Information about indices}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Relational Representation of System Metadata}
    \begin{block}{}
        \begin{itemize}
            \item Relational representation on disk
            \item Specialized data structures designed for efficient access, in memory
        \end{itemize}
    \end{block}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{Relation\_metadata}
                relation\_name, number\_of\_attributes, storage\_organization, location
            \end{exampleblock}
            \begin{exampleblock}{Index\_metadata}
                index\_name, relation\_name, index\_type, index\_attributes
            \end{exampleblock}
            \begin{exampleblock}{View\_metadata}
                view\_name, definition
            \end{exampleblock}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{Attribute\_metadata}
                relation\_name, attribute\_name, domain\_type, position, length
            \end{exampleblock}
            \begin{exampleblock}{User\_metadata}
                user\_name, encrypted\_password, group
            \end{exampleblock}
        \end{column}
    \end{columns}
\end{frame}

\section{Storage Access}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Storage Access\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Storage Access}
    \begin{block}{}
        \begin{itemize}
            \item A database file is partitioned into fixed-length storage units called blocks
            \item Blocks are units of both storage allocation and data transfer
            \item Database system seeks to minimize the number of block transfers between the disk and memory
            \item We can reduce the number of disk accesses by keeping as many blocks as possible in main memory
            \item \textbf{Buffer} : portion of main memory available to store copies of disk blocks
            \item \textbf{Buffer Manager} : subsystem responsible for allocating buffer space in main memory
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Buffer Manager}
    \begin{block}{}
        \begin{itemize}
            \item Programs call on the buffer manager when they need a block from disk
            \item If the block is already in the buffer, buffer manager returns the address of the block in main memory
            \item If the block is not in the buffer, the buffer manager
            \begin{itemize}
                \item Allocates space in the buffer for the block
                \item Replacing (throwing out) some other block, if required, to make space for the new block
                \item Replaced block written back to disk only if it was modified since the most recent time that it was written to / fetched from the disk
                \item Reads the block from the disk to the buffer, and returns the address of the block in main memory to requester
            \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Buffer Replacement Policies}
    \begin{block}{}
        \begin{itemize}
            \item Most operating systems replace the block least recently used (\textbf{LRU strategy})
            \item Idea behind LRU - use past pattern of block references as a predictor of future references
            \item Queries have well-defined access patterns (such as sequential scans), and a database system can use the information in a user's query to predict future references
            \item LRU may be a bad strategy for certain access patterns involving repeated scans of data
            \item Mixed strategy with hints on replacement strategy provided by the query optimizer is preferable
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Buffer Replacement Policies (2)}
    \begin{block}{}
        \begin{itemize}
            \item \textbf{Pinned block} : memory block that is not allowed to be written back to disk
            \item \textbf{Toss-immediate strategy} : frees the space occupied by a block as soon as the final tuple of that block has been processed
            \item \textbf{Most recently used (MRU) strategy} : system must pin the block currently being processed. After the final tuple has been processed, the block is unpinned, and it becomes the most recently used block.
            \item Buffer manager can use statistical information regarding the probability that a request will reference a particular relation (e.g. data dictionary is frequently accessed, so keep it in buffer)
            \item Buffer managers also support forced output of blocks for the purpose of recovery
        \end{itemize}
    \end{block}
\end{frame}
