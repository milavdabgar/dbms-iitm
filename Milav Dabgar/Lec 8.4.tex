\section{Physical Storage Media}

\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Physical Storage Media\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Classification of Physical Storage Media}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Speed with which data can be accessed
            \item Cost per unit of data
            \item Reliability
            \begin{itemize}[<+->]
                \item data loss on power failure or system crash
                \item physical failure of the storage device
            \end{itemize}
            \item Can differentiate storage into:
            \begin{itemize}[<+->]
                \item \textbf{volatile storage} : loses contents when power is switched off
                \item \textbf{non-volatile storage} : Contents persist even when power is switched off. Includes secondary and tertiary storage, as well as battery-backed up main-memory
            \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Physical Storage Media}
    \begin{block}{Cache}
        \begin{itemize}[<+->]
            \item fastest and most costly form of storage
            \item volatile
            \item managed by the computer system hardware
        \end{itemize}
    \end{block}
    \begin{block}{Main memory}
        \begin{itemize}[<+->]
            \item fast access (10's to 100's of nanoseconds)
            \item generally too small (or too expensive) to store the entire database
            \item capacities of up to a few Gigabytes widely used currently
            \item Capacities have gone up and per-byte costs have decreased steadily and rapidly
        \end{itemize}
    \end{block}
    \begin{alertblock}{Volatile}
        contents of main memory are usually lost if a power failure or system crash occurs
    \end{alertblock}
\end{frame}

\begin{frame}{Physical Storage Media (2): Flash memory}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Data survives power failure
            \item Data can be written at a location only once, but location can be erased and written to again
            \item Can support only a limited number (10K - 1M) of write/erase cycles
            \item Erasing of memory has to be done to an entire bank of memory
            \item Reads are roughly as fast as main memory
            \item But writes are slow (few microseconds), erase is slower
            \item Widely used in embedded devices such as digital cameras, phones, and USB keys
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Physical Storage Media (3): Magnetic Disk}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Data is stored on spinning disk, and read/written magnetically
            \item Primary medium for the long-term storage of data
            \item typically stores entire database
            \item Data must be moved from disk to main memory for access, and written back for storage - much slower access than main memory
            \item direct-access
            \item possible to read data on disk in any order, unlike magnetic tape
            \item Capacities range up to roughly 16-32 TB
            \item Much larger capacity and much lower cost/byte than main memory/flash memory
            \item Growing constantly and rapidly with technology improvements
            \item Survives power failures and system crashes (disk failure can destroy data, but is rare)
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Physical Storage Media (4): Optical Storage}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item non-volatile, data is read optically from a spinning disk using a laser
            \item CD-ROM (640 MB) and DVD (4.7 to 17 GB) most popular forms
            \item Blu-ray disks: 27 GB to 54 GB
            \item Write-one, read-many (WORM) optical disks used for archival storage (CD-R, DVD-R, DVD+R)
            \item Multiple write versions also available (CD-RW, DVD-RW, DVD+RW, and DVD-RAM)
            \item Reads and writes are slower than with magnetic disk
            \item Juke-box systems, with large numbers of removable disks, a few drives, and a mechanism for automatic loading/unloading of disks available for storing large volumes of data
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Physical Storage Media (5): Tape Storage}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item non-volatile, used primarily for backup (to recover from disk failure), and for archival data
            \item sequential-access
            \item much slower than disk
            \item very high capacity (40 to 300 TB tapes available)
            \item tape can be removed from drive storage costs much cheaper than disk, but drives are expensive
            \item Tape jukeboxes available for storing massive amounts of data: hundreds of terabytes (TB) to multiple petabytes (PB)
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Magnetic Disk (2): Mechanism}
    \begin{columns}
        \begin{column}{0.65\textwidth}
            \begin{block}{}
                \begin{itemize}[<+->]
                    \item Read-write head is positioned very close to the platter surface, reads/writes magnetically encoded information.
                    \item Surface of platter divided into circular tracks (Over 50K-100K tracks per platter).
                    \item Each track is divided into sectors. A sector is the smallest unit of data read or written (typically 512 bytes).
                    \item To read/write a sector: disk arm swings to position head on right track, platter spins: Read/Write as sector passes under head.
                    \item \textbf{Head-disk assemblies}: multiple disk platters on a single spindle (1 to 5 usually), one head per platter, mounted on a common arm.
                    \item \textbf{Cylinder} $i$ consists of $i$-th track of all the platters
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.35\textwidth}
            \begin{center}
                \includegraphics[height=0.7\textheight, width=\textwidth, keepaspectratio]{"../Lecture 8.4_artifacts/image_000017_d58b506142eadc48ef8dbe902cc64794b2b4b4624fe54ac9542e89036f2ece6c.png"}
            \end{center}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Magnetic Disks (3): Disk Controller, Subsystems, and Interfaces}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Disk Controller} : interfaces between the computer system and the disk drive hardware
            \begin{itemize}[<+->]
                \item Accepts high-level commands to read or write a sector
                \item Initiates actions moving the disk arm to the right track, reading or writing the data
                \item Computes and attaches checksums to each sector to verify that correct read back
                \item Ensures successful writing by reading back sector after writing it
                \item Performs remapping of bad sectors
            \end{itemize}
            \item \textbf{Disk Subsystem} :
            \begin{itemize}[<+->]
                \item Disk Interface Standards Families : ATA, SATA, SCSI, SAS, several variants
                \item Storage Area Networks (SAN) connects disks by a high-speed network to a number of servers
                \item Network Attached Storage (NAS) provides a file system interface using networked file system protocol
            \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Magnetic Disks (4): Performance Measures}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Access Time} : time from a read or write request issue to start of data transfer:
            \begin{itemize}[<+->]
                \item \textbf{Seek Time} : time to reposition the arm over the correct track (4 to 10 ms)
                \item \textbf{Rotational Latency} : time for the sector to be accessed to appear under the head (4 to 11 ms)
            \end{itemize}
            \item \textbf{Data-transfer Rate} : the rate at which data can be retrieved from or stored to the disk (25 to 100 MB per second max rate)
            \item \textbf{Mean Time To Failure (MTTF)} : Avg. time the disk is expected to run continuously without any failure
            \begin{itemize}[<+->]
                \item Typically 3 to 5 years (theoretical MTTF of 500,000 to 1,200,000 hours)
                \item MTTF decreases as disk ages
            \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Magnetic Tapes}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Hold large volumes of data and provide high transfer rates
            \item Tape Formats: DAT, DLT, Ultrium, Ampex helical scan format
            \item Transfer rates from few to 10's of MB/s
            \item Tapes are cheap, but cost of drives is very high
            \item Very slow access time in comparison to magnetic and optical disks (Limited to sequential access)
            \item Used mainly for backup, for storage of infrequently used information, and as an off-line medium
            \item Tape jukeboxes used for very large capacity storage (Multiple petabyes)
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Cloud Storage}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Cloud storage is purchased from a third-party cloud vendor who owns and operates data storage capacity and delivers it over the Internet in a pay-as-you-go model
            \item These cloud storage vendors manage capacity, security and durability to make data accessible to applications all around the world
            \item Applications access cloud storage through traditional storage protocols or directly via an API
            \item Many vendors offer complementary services designed to help collect, manage, secure and analyze data at massive scale. Various available options for cloud storage are:
            \begin{itemize}[<+->]
                \item Google Drive, Amazon Drive, OneDrive, Evernote, Dropbox, etc.
            \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Optical Disks}
    \begin{block}{Compact disk-read only memory (CD-ROM)}
        \begin{itemize}[<+->]
            \item Removable disks, 640 MB per disk
            \item Seek time about 100 msec (optical read head is heavier and slower)
            \item Higher latency (3000 RPM) and lower data-transfer rates (3-6 MB/s) compared to magnetic disks
        \end{itemize}
    \end{block}
    \begin{block}{Digital Video Disk (DVD) \& Blu-ray}
        \begin{itemize}[<+->]
            \item DVD-5 holds 4.7 GB, and DVD-9 holds 8.5 GB
            \item DVD-10 and DVD-18 are double sided formats with capacities of 9.4 GB and 17 GB
            \item Blu-ray DVD: 27 GB (54 GB for double sided disk)
            \item Record once versions (CD-R and DVD-R) are popular (high capacity and long lifetime; archival)
            \item Multi-write versions (CD-RW, DVD-RW, DVD+RW and DVD-RAM) also available
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Flash Drives}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Flash drives are often referred to as pen drives, thumb drives, or jump drives. They have completely replaced floppy drives for portable storage.
            \item They have nearly replaced CDs and DVDs for data storage purposes.
            \item USB flash drives are removable and rewritable storage devices that require a USB port for connection and utilizes non-volatile flash memory technology.
            \item The storage space in USB flash drives is quite large with sizes ranging from 128MB to 2TB.
            \item The USB standard a flash drive is built around will determine the number of things about its potential performance, including maximum transfer rate.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Secure Digital Cards (SD cards)}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item A Secure Digital (SD, in short) card is a type of removable memory card used to read and write large quantities of data.
            \item Widely used in mobile electronics, cameras, smart devices, video game consoles.

\begin{frame}{Flash Storage}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item NOR flash vs NAND flash
            \item \textbf{NAND flash}:
            \begin{itemize}[<+->]
                \item used widely for storage, since it is much cheaper than NOR flash
                \item requires page-at-a-time read (page: 512 bytes to 4 KB)
                \item transfer rate around 20 MB/sec
            \end{itemize}
            \item \textbf{solid state disks} : Use multiple flash storage devices to provide higher transfer rate of 100 to 200 MB/sec
            \item erase is very slow (1 to 2 ms)
            \begin{itemize}[<+->]
                \item erase block contains multiple pages
                \item remapping of logical page addresses to physical page addresses avoids waiting for erase (translation table tracks mapping)
                \item after 100,000 to 1,000,000 erases, erase block becomes unreliable (wear leveling)
            \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Solid-State Drives (SSD)}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item SSDs replace traditional mechanical hard disks by using flash-based memory, which is significantly faster.
            \item SSDs speed up computers significantly due to their low read-access times and fast throughput.
            \item The idea of SSDs was introduced in 1978. It stores the data in the persistent state even when no power is supplied.
            \item The speed of SSD is much larger than that of HDD as it reads/writes data at higher input-output per second.
            \item Unlike HDDs, SSDs do not include any moving parts. SDDs can resist vibrations and high temperatures.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Future of Storage: DNA Digital Storage}
    \begin{block}{}
        \begin{itemize}
            \item DNA digital data storage is the process of encoding and decoding binary data to and from synthesized strands of DNA.
            \item While DNA has enormous potential because of its high storage density, its practical use is currently severely limited because of its high cost and very slow read and write times.
            \item This information can be encoded in DNA using the four nucleotides: A, T, G, and C. For example, G and C could define 0 while A and T represent 1.
            \item DNA is extremely stable and is fairly easy (but expensive) to synthesize and sequence.
            \item High density - each nucleotide is about 1 cubic nanometer - an exabyte ($10^{18}$ bytes) of data stored as DNA could fit in the palm of your hand.
            \item DNA Synthesis: A DNA synthesizer machine builds synthetic DNA strands matching the sequence of digital code.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Future of Storage: Quantum Memory}
    \begin{block}{}
        \begin{itemize}
            \item Quantum memory is the quantum-mechanical version of ordinary computer memory.
            \item Whereas ordinary memory stores information as binary states, quantum memory stores a quantum state for later retrieval.
            \item These states hold useful computational information known as qubits.
            \item Quantum memory is essential for the development of many devices in quantum information processing applications such as quantum network, quantum repeater, linear optical quantum computation or long-distance quantum communication.
            \item Unlike classical memory, the states stored in quantum memory can be in a quantum superposition, giving much more practical flexibility in quantum algorithms.
        \end{itemize}
    \end{block}
\end{frame}
