\begin{frame}{Python psycopg2 Module APIs: connection objects}
    \begin{itemize}[<+->]
        \item \texttt{psycopg2.connect(database, user, password, host, port)}
        \begin{itemize}
            \item[$\triangleright$] Opens a connection to the PostgreSQL database; returns a connection object if successful
        \end{itemize}
        \item \texttt{connection.close()}
        \begin{itemize}
            \item[$\triangleright$] Closes the database connection
        \end{itemize}
        \item \texttt{connection.cursor()}
        \begin{itemize}
            \item[$\triangleright$] Creates a cursor which will be used throughout the program
        \end{itemize}
        \item \texttt{cursor.close()}
        \begin{itemize}
            \item[$\triangleright$] Closes the cursor
        \end{itemize}
    \end{itemize}
    \vspace{1em}
    \small Source: \url{https://www.tutorialspoint.com/postgresql/postgresql_python.htm}
\end{frame}

\begin{frame}{Python psycopg2 Module APIs: insert, delete, update \& stored procedures}
    \begin{itemize}
        \item \texttt{cursor.execute(sql [, optional parameters])}
        \begin{itemize}
            \item This routine executes an SQL statement. The SQL statement may be parameterized (i.e., placeholders instead of SQL literals). The psycopg2 module supports placeholder using \texttt{\%s} sign. For example: \texttt{cursor.execute("insert into people values (\%s, \%s)", (who, age))}
        \end{itemize}
        \item \texttt{cursor.executemany(sql, seq of parameters)}
        \begin{itemize}
            \item This routine executes an SQL command against all parameter sequences or mappings found in the sequence SQL.
        \end{itemize}
        \item \texttt{cursor.callproc(procname[, parameters])}
        \begin{itemize}
            \item This routine executes a stored database procedure with the given name. The sequence of parameters must contain one entry for each argument that the procedure expects.
        \end{itemize}
        \item \texttt{cursor.rowcount}
        \begin{itemize}
            \item This is a read-only attribute which returns the total number of database rows that have been modified, inserted, or deleted by the last \texttt{execute()}.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Python psycopg2 Module APIs: select}
    \begin{itemize}
        \item \texttt{cursor.fetchone()}
        \begin{itemize}
            \item This method fetches the next row of a query result set, returning a single sequence, or None when no more data is available.
        \end{itemize}
        \item \texttt{cursor.fetchmany([size=cursor.arraysize])}
        \begin{itemize}
            \item This routine fetches the next set of rows of a query result, returning a list. An empty list is returned when no more rows are available. The method tries to fetch as many rows as indicated by the size parameter.
        \end{itemize}
        \item \texttt{cursor.fetchall()}
        \begin{itemize}
            \item This routine fetches all (remaining) rows of a query result, returning a list. An empty list is returned when no rows are available.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Python psycopg2 Module APIs: commit \& rollback}
    \begin{itemize}[<+->]
        \item \texttt{connection.commit()}
        \begin{itemize}
            \item[$\triangleright$] Commits the current transaction. Changes not committed are not visible to other database connections.
        \end{itemize}
        \item \texttt{connection.rollback()}
        \begin{itemize}
            \item[$\triangleright$] Rolls back any changes to the database since the last call to \texttt{commit()}.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Connect to a PostgreSQL Database Server}

import psycopg2
def connectDb(dbname, usrname, pwd, address, portnum):
  conn = None
  try: # connect to the PostgreSQL database
    conn = psycopg2.connect(database = dbname, user = usrname, \
                            password = pwd, host = address, port = portnum)
    print ("Database connected successfully")
  except (Exception, psycopg2.DatabaseError) as error:
    print(error)
  finally:
    if conn is not None:
      conn.close() # close the connection

connectDb("mydb", "myuser", "mypass", "127.0.0.1", "5432") # function call
    \end{verbatim}
    \textbf{Output:}
    \texttt{Database connected successfully}
    \begin{itemize}
        \item \texttt{psycopg2.DatabaseError} : Exception raised for errors that are related to the PostgreSQL database.
        \item We assume the following for all the programs in this module:
        \begin{itemize}
            \item Database Name: \texttt{mydb}, Username: \texttt{myuser}, Password: \texttt{mypass}, Host Name: \texttt{localhost} or IP address \texttt{127.0.0.1}
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Steps to execute SQL commands}
    \begin{enumerate}
        \item Use the \texttt{psycopg2.connect()} method with the required arguments to connect PostgresSQL. It would return an Connection object if the connection established successfully.
        \item Create a cursor object using the \texttt{cursor()} method of connection object.
        \item The \texttt{execute()} methods run the SQL commands and return the result.
        \item Use \texttt{cursor.fetchall()} or \texttt{fetchone()} or \texttt{fetchmany()} to read query result.
        \item Use \texttt{commit()} to make the changes in database persistent, or use \texttt{rollback()} to revert the database changes.
        \item Use \texttt{cursor.close()} and \texttt{connection.close()} method to close the cursor and PostgreSQL connection.
    \end{enumerate}
    \vspace{1em}
    \small Source: \url{https://pynative.com/python-postgresql-tutorial/}
\end{frame}

\begin{frame}[fragile]{CREATE new PostgreSQL tables}
    \begin{verbatim}
import psycopg2
def createTable():
  conn = None
  try:
    conn = psycopg2.connect(database="mydb", user="myuser", password="mypass", host="127.0.0.1", port="5432")
    cur = conn.cursor()
    cur.execute('''CREATE TABLE EMPLOYEE \
        (emp_num INT PRIMARY KEY NOT NULL, \
        emp_name VARCHAR(40) NOT NULL, \
        department VARCHAR(40) NOT NULL)''') # execute the CREATE TABLE statement
    conn.commit() # commit the changes to the database
    print ("Table created successfully")
    cur.close() # close the cursor
  except (Exception, psycopg2.DatabaseError) as error:
    print(error)
  finally:
    if conn is not None: conn.close() # close the connection
createTable() #function call
    \end{verbatim}
    \textbf{Output (if table EMPLOYEE does not exist):} \texttt{Table created successfully} \\
    \textbf{Output (if table EMPLOYEE already exists):} \texttt{relation "employee" already exists}
\end{frame}

import psycopg2
def insertRecord(num, name, dept):
  conn = None
  try:
    conn = psycopg2.connect(database="mydb", user="myuser", password="mypass", host="127.0.0.1", port="5432")

\begin{frame}{Flask Web Application Framework}
    \begin{itemize}[<+->]
        \item Flask is a lightweight WSGI web application framework -- quick to start, scales to complex applications
        \item Built around Werkzeug (WSGI toolkit) and Jinja (template engine)
        \item Flask offers suggestions but does not enforce any dependencies or project layouts
        \item There are many community extensions that make adding new functionality easy

\begin{frame}[fragile]{A Simple Example}
    \begin{verbatim}
from flask import Flask
app = Flask(__name__)

@app.route('/')
def hello_world():
  return "Hello World"

if __name__ == '__main__':
  app.run()
    \end{verbatim}
    \begin{itemize}
        \item Importing flask module in the project is mandatory. Our WSGI application is an object of \texttt{Flask} class.
        \item Flask constructor takes the name of current module (\texttt{\_\_name\_\_}) as argument.
        \item The \texttt{route()} function of the Flask class is a decorator, which tells the application which URL should call the associated function. \texttt{app.route(rule, options)}
        \begin{itemize}
            \item The \texttt{rule} parameter represents URL binding with the function.
            \item The \texttt{options} is a list of parameters to be forwarded to the underlying Rule object.
        \end{itemize}
        \item In the above example, \texttt{'/'} URL is bound with \texttt{hello\_world()} function. Hence, when the home page of web server is opened in browser, the output of this function will be rendered.
        \item Finally the \texttt{run()} method of Flask class runs the application on the local development server.
    \end{itemize}
\end{frame}
