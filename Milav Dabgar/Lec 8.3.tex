\section{Data Structure}

\begin{frame}
  \vfill
  \centering

\begin{frame}{Data Structure}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Data structure} : A data structure specifies the way of organizing and storing in-memory data that enables efficient access and modification of the data.
            \item Linear Data Structures vs Non-linear Data Structures
            \item Most data structure has a container for the data and typical operations that its needs to perform
            \item For applications relating to data management, the key operations are:
            \begin{itemize}[<+->]
                \item Create
                \item Insert
                \item Delete
                \item Find / Search
                \item Close
            \end{itemize}
            \item Efficiency is measured in terms of time and space taken for these operations
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Non-linear Data Structures: Why?}
    \begin{block}{Summary observations for linear structures:}
        \begin{itemize}[<+->]
            \item All of them have the space complexity $\mathcal{O}(n)$, which optimal. However, the actual used space may be lower in array while linked list has an overhead of 100\% (double)
            \item All of them offer satisfactory complexity for some operations while being unsatisfactory on the others.
        \end{itemize}
    \end{block}
    \begin{table}[]
        \centering
        \begin{tabular}{lllll}
        \toprule
                & \textbf{Array} & \textbf{Array} & \textbf{Linked List} & \textbf{Linked List} \\
                & \textit{Unordered} & \textit{Ordered} & \textit{Unordered}   & \textit{Ordered}     \\ \midrule
        Access  & $\mathcal{O}(1)$ & $\mathcal{O}(1)$ & $\mathcal{O}(n)$       & $\mathcal{O}(n)$       \\
        Insert  & $\mathcal{O}(n)$ & $\mathcal{O}(n)$ & $\mathcal{O}(1)$       & $\mathcal{O}(1)$       \\
        Delete  & $\mathcal{O}(n)$ & $\mathcal{O}(n)$ & $\mathcal{O}(1)$       & $\mathcal{O}(1)$       \\
        Search  & $\mathcal{O}(n)$ & $\mathcal{O}(\log n)$ & $\mathcal{O}(n)$       & $\mathcal{O}(n)$       \\ \bottomrule
        \end{tabular}
    \end{table}
    \begin{alertblock}{Non-Linear data structures \dots}
        can be used to trade-off between extremes and achieve a balanced good performance for all.
    \end{alertblock}
\end{frame}

\begin{frame}{Non-linear Data Structures (2)}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Nonlinear data structures are those data structures in which data items are not arranged in a sequence and each element may have multiple paths to connect to other elements.
            \item Traversing through the elements is not possible in one run as the data is non-linearly arranged.
            \item Common Non-Linear Data Structures include:
            \begin{itemize}[<+->]
                \item \textbf{Graph} : Undirected or Directed, Unweighted or Weighted, and variants
                \item \textbf{Tree} : Rooted or Unrooted, Binary or n-ary, Balanced or Unbalanced, and variants
                \item \textbf{Hash Table} : Array with lists (coalesced chains) and one or more hash functions
                \item \textbf{Skip List} : Multi-layered interconnected linked lists
            \end{itemize}
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Non-linear Data Structures (3): Graph}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Graphs} : Graph $G$ is a collection of vertices $V$ (store the elements) and connecting edges (links) $E$ between vertices: $G = \langle V, E \rangle$ where $E \subseteq V \times V$
            \item A graph may be: Undirected or Directed, Unweighted or Weighted, Cyclic or Acyclic, Disconnected or Connected.
            \item Examples of a graph include: ER Diagram, Networks (Electrical, Water), Friendships in Facebook, Knowledge Graph.
        \end{itemize}
    \end{block}

\begin{frame}{Non-linear Data Structures (8): Tree}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{}
                \begin{itemize}[<+->]
                    \item \textbf{Fact 1} : A tree with $n$ nodes has $n - 1$ edges
                    \item \textbf{Fact 2} : The maximum number of nodes at level $l$ of a binary tree is $2^l$.
                    \item \textbf{Fact 3} : If $h$ is the height of a binary tree of $n$ nodes, then:
                    \begin{itemize}[<+->]
                        \item $h + 1 \leq n \leq 2^{h + 1} - 1$
                        \item $\lceil \lg(n + 1) \rceil - 1 \leq h \leq n - 1$
                        \item $\mathcal{O}(\log n) \leq h \leq \mathcal{O}(n)$
                    \end{itemize}
                    \item For a $k$-ary tree, $\mathcal{O}(\log_k n) \leq h \leq \mathcal{O}(n)$
                \end{itemize}
            \end{block}
        \end{column}

\section{Binary Search Tree}

\begin{frame}
  \vfill

  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Binary Search and Binary Search Tree}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item During the study of linear data structure, we observed that:
            \begin{itemize}[<+->]
                \item Binary search is efficient in search of a key: $\mathcal{O}(\log n)$. However, it needs to be performed on a sorted array, and the array makes insertion and deletion expensive at $\mathcal{O}(n)$.
                \item The linked list is efficient in insertion and deletion at $\mathcal{O}(1)$, while it makes the search expensive at $\mathcal{O}(n)$.
            \end{itemize}
            \item Using non-linearity, specifically (binary) trees, we can combine the benefits of both.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Binary Search and Binary Search Tree (2)}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Note that once an array is sorted, we know the order in which its elements may be checked (for any key) during a search.
            \item As the binary search splits the array, we can conceptually consider the Middle Element to be the Root of a tree and the left/right sub-array to be its left/right sub-tree.
            \item Progressing recursively, we have a Binary Search Tree.
        \end{itemize}
    \end{block}
    \begin{center}
        \includegraphics[height=0.3\textheight, width=\textwidth, keepaspectratio]{"../Lecture 8.3_artifacts/image_000027_ae3cb3965e1f2fe25272a09ab1c46410e97e5395d595117516823a3bc83765e5.png"}
    \end{center}
\end{frame}

\begin{frame}{Binary Search Tree}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Binary Search Tree (BST)} : Is a tree in which all the nodes hold the following:
            \begin{itemize}[<+->]
                \item The value of each node in the left sub-tree is less than the value of its root.
                \item The value of each node in the right sub-tree is greater than the value of its root.
            \end{itemize}
            \item \textbf{Structure of BST node} : Each node consists of an element, a link to the left child or the left subtree, and a link to the right child or the right subtree.
        \end{itemize}
    \end{block}
    \begin{center}
        \includegraphics[height=0.35\textheight, width=\textwidth, keepaspectratio]{"../Lecture 8.3_artifacts/image_000029_81267e4f3b30ea8615f88d3684d106e7d98b8311cf170b50c9acf8392fb13ae8.png"}

\begin{frame}{Searching a key in BST}
    \begin{block}{search(root, key)}
        \begin{enumerate}
            \item Compare the key with the element at root.
            \item If the key is equal to root's element then Element found and return
            \item else if the key is lesser than the root's element, \texttt{search(root.lc)} \#search on the left subtree
            \item else: \texttt{search(root.rc)} \#search on the right subtree
        \end{enumerate}
    \end{block}
    \begin{center}
        \includegraphics[height=0.3\textheight, width=\textwidth, keepaspectratio]{"../Lecture 8.3_artifacts/image_000035_6183bd0bbe870e66bd5355275895f317e993c762fd21dd4f7e4aa81b6335836d.png"}
    \end{center}
\end{frame}

\begin{frame}{Searching a key in BST (2)}
    \begin{block}{}
        Searching a key in a BST is $\mathcal{O}(h)$, where $h$ is the height of the tree.
    \end{block}
    \begin{alertblock}{Worst Case}
        \begin{itemize}[<+->]
            \item The BST is a skewed binary search tree (can happen if keys are inserted in sorted order).
            \item Height ($h$) of the BST having $n$ elements becomes $n - 1$.
            \item Time complexity of search in BST becomes $\mathcal{O}(n)$.
        \end{itemize}
    \end{alertblock}
    \begin{exampleblock}{Best Case}
        \begin{itemize}[<+->]
            \item The BST is a balanced binary search tree (possible if keys are inserted in purely randomized order, or the tree is explicitly balanced).
            \item Height ($h$) of the binary search tree becomes $\log n$.
            \item Time complexity of search in BST becomes $\mathcal{O}(\log n)$.
        \end{itemize}
    \end{exampleblock}
\end{frame}
