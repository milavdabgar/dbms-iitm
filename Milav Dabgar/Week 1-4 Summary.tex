\begin{frame}{Module 6 Recap}
    \begin{itemize}[<+->]
        \item \textbf{Attribute Types} - The set of allowed values for each attribute is called the domain of the attribute. (i.e. Alphanumeric string, Alpha string, Date, number etc.)
        \item \textbf{Schema} - $R = (A_1, A_2, \cdots, A_n)$ is a relation schema where $A_1, A_2, \cdots, A_n$ are attributes.
        \item \textbf{Instances} : The collection of information stored in the database at a particular moment is called an instance of the database.
        \item Let $K \subseteq R$, where $R$ is the set of attributes in the relation.
        \item $K$ is a \textbf{super key} of $R$ if values for $K$ are sufficient to identify a unique tuple of each possible relation $r(R)$.
        \item Super key $K$ is a \textbf{candidate key} if $K$ is minimal.
        \item One of the candidate keys is selected to be the \textbf{primary key}.
        \item A \textbf{surrogate key} (or synthetic key) in a database is a unique identifier for either an entity in the modeled world or an object in the database.
    \end{itemize}
\end{frame}

\begin{frame}{Module 6 Recap (Cont.)}
    \begin{itemize}[<+->]
        \item \textbf{Secondary / Alternate Key} : candidate keys other than primary key.
        \item \textbf{Simple key} : Consists of a single attribute.
        \item \textbf{Composite key} : Consists of more than one attribute to uniquely identify each tuples in a relation.
        \item \textbf{Foreign key} : Value in one relation must appear in another.
        \item \textbf{Compound key} : consists of more than one attribute to uniquely identify an entity occurrence.
        \item \textbf{Relational Query language} :
        \begin{itemize}
            \item[$\triangleright$] \textbf{Procedural programming} : requires that the programmer tell the computer what to do. That is, how to get the output for the range of required inputs.
            \item[$\triangleright$] \textbf{Declarative programming} : requires a more descriptive style. The programmer must know what relationships hold between various entities
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Module 7 Recap}
    \begin{itemize}[<+->]
        \item \textbf{select operation} : selection of rows(tuples). i.e: $\sigma_{D > 5}(r)$
        \item \textbf{Project operation} : selection of columns (Attributes). i.e. $\pi_{A,C}(r)$
        \item \textbf{Union} : union of two relations. i.e. $r \cup s$
        \item \textbf{Difference} : Set difference of two relations. i.e. $r - s$
        \item \textbf{Intersection} : Set intersection of two relations. i.e. $r \cap s$
        \item \textbf{Cartesian Product} : Joining two relations - Cartesian-product. i.e. $r \times s$
        \item \textbf{Natural Join} : natural join operation on two relations matches tuples whose values are the same on all attribute names that are common to both relations. i.e. $r \bowtie s$
    \end{itemize}
\end{frame}

\begin{frame}{Module 8 Recap}
    \begin{itemize}[<+->]
        \item History of SQL
        \item \textbf{Data Definition Language (DDL)} : The SQL DDL provides commands for defining relation schemas, deleting relations, and modifying relation schema. (i.e. \texttt{CREATE TABLE}, \texttt{DROP TABLE}, \texttt{ALTER} etc.)
        \item \textbf{Data Manipulation Language (DML)} : The SQL DML provides the ability to query information from the database and to insert tuples into, delete tuples from, and modify tuples in the database. (i.e. \texttt{UPDATE}, \texttt{INSERT}, \texttt{DELETE} etc.)
        \item \textbf{Basic SQL structure}: \texttt{SELECT $A_1, A_2, \dots , A_n$ FROM $r_1, r_2, \dots , r_m$ WHERE $P$}
        \begin{itemize}
            \item[$\triangleright$] \textbf{Select clause}: lists the attributes desired in the result of a query.
            \item[$\triangleright$] \textbf{Where clause} : specifies conditions that the result must satisfy.
            \item[$\triangleright$] \textbf{From clause} : lists the relations involved in the query.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Module 9 Recap}
    \begin{itemize}[<+->]
        \item \textbf{Cartesian Product} : The Cartesian product instructor X teaches. It generates every possible instructor-teaches pair, with all attributes from both relations. 
        \begin{exampleblock}{SQL query}
            \begin{verbatim}select * from instructor, teaches\end{verbatim}
        \end{exampleblock}
        \item \textbf{Rename AS Operation} : The SQL allows renaming relations and attributes using the \texttt{as} clause. (i.e. \texttt{old\_name as new\_name})
        \item \textbf{String Operation} : SQL includes a string-matching operator for comparisons on character strings. The operator \texttt{like} uses patterns that are described using two special characters:
        \begin{itemize}
            \item[$\triangleright$] percent (\texttt{\%}). The \texttt{\%} character matches any substring
            \item[$\triangleright$] underscore (\texttt{\_}). The \texttt{\_} character matches any single character
        \end{itemize}
        \item \textbf{Order By} : We may specify \texttt{desc} for descending order or \texttt{asc} for ascending order, for each attribute; ascending order is the default. (i.e. \texttt{order by name desc})
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Module 9 Recap (Cont.)}
    \begin{itemize}[<+->]
        \item \textbf{Select top/ Fetch} : The \texttt{Select Top} clause is used to specify the number of records to return. 
        \begin{exampleblock}{SQL Server Example}
            \begin{verbatim}select top 10 distinct name from instructor\end{verbatim}
        \end{exampleblock}
        \uncover<2->{

fetch first 10 rows only\end{verbatim}
        \end{exampleblock}
        \item \textbf{Where Clause Predicate} : SQL includes a \texttt{between} comparison operator, Tuple comparison.
        \item \textbf{IN operator} : The \texttt{in} operator allows you to specify multiple values in a where clause.
        \item \textbf{Duplicates} : In relations with duplicates, SQL can define how many copies of tuples appear in the result. Multiset versions of some relational algebra operators.
    \end{itemize}
\end{frame}

\begin{frame}{Module 10 Recap}
    \begin{itemize}[<+->]
        \item \textbf{Set Operations} : \texttt{union}, \texttt{intersect}, \texttt{except}
        \begin{itemize}
            \item[$\triangleright$] Each of the above operations automatically eliminates duplicates.
            \item[$\triangleright$] To retain all duplicates use the corresponding multiset versions \texttt{union all}, \texttt{intersect all}, and \texttt{except all}.
        \end{itemize}
        \item \textbf{Null Values} : \texttt{null} signifies an unknown value or that a value does not exist.
        \item \textbf{Aggregate Functions} :
        \begin{itemize}
            \item[$\triangleright$] \texttt{avg}: return average value
            \item[$\triangleright$] \texttt{min}: return min value
            \item[$\triangleright$] \texttt{max}: return max value
            \item[$\triangleright$] \texttt{sum}: return sum of values
            \item[$\triangleright$] \texttt{count}: return number of values
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Module 10 Recap (Cont.)}
    \begin{itemize}[<+->]
        \item \textbf{Group By} :
        \begin{itemize}
            \item[$\triangleright$] The attribute or attributes given in the \texttt{group by} clause are used to form groups.
            \item[$\triangleright$] Tuples with the same value on all attributes in the \texttt{group by} clause are placed in one group.
        \end{itemize}
        \item \textbf{Having} : predicates in the \texttt{having} clause are applied after the formation of groups.
        \item \textbf{Null Values} : All aggregate operations except \texttt{count(*)} ignore tuples with null values on the aggregated attributes.
    \end{itemize}
\end{frame}

\begin{frame}{Tuple Relational Calculus (TRC)}
    \begin{itemize}[<+->]
        \item TRC is a nonprocedural query language, where each query is of the form
        \item[] $\{ t \mid P(t) \}$
        \item \textbf{where} $t$ = resulting tuples,
        \item $P(t)$ = known as predicate and these are the conditions that are used to fetch $t$.
        \item $P(t)$ may have various conditions logically combined with OR ($\lor$), AND ($\land$), NOT($\neg$).
        \item It also uses \textbf{quantifiers}:
        \begin{itemize}
            \item[$\triangleright$] $\exists\ t \in r\ (Q(t))$ = `there exists' a tuple in $t$ in relation $r$ such that predicate $Q(t)$ is true.
            \item[$\triangleright$] $\forall\ t \in r\ (Q(t))$ = $Q(t)$ is true `for all' tuples in relation $r$.
        \end{itemize}
        \item Example: $\{ P \mid \exists\ S \in \text{Students}\ (S.\text{CGPA} > 8 \land P.\text{name} = S.\text{sname} \land P.\text{age} = S.\text{age}) \}$
        \begin{itemize}
            \item[$\triangleright$] Returns the name and age of students with a CGPA above 8.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Domain Relational Calculus (DRC)}
    \begin{itemize}[<+->]
        \item $\{ \langle x_1, x_2, \dots, x_n \rangle \mid P(x_1, x_2, \dots, x_n) \}$
        \item $x_1, x_2, \dots, x_n$ represent domain variables
        \item $P$ represents a formula similar to that of the predicate calculus
        \item Example Relation Students (Name, Age, Marks, Subject)
        \item Example Query: $\{ \langle a, b \rangle \mid \exists\ a,b,c,d (\langle a,b,c,d \rangle \in \text{students} \land c > 75) \}$
        \begin{itemize}
            \item[$\triangleright$] Returns the name and age of students having marks above 75.
        \end{itemize}
        \item \textbf{Note}: You have to mention the domain variables in the same order as given in the table.
    \end{itemize}
\end{frame}
