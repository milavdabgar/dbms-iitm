\section{Transaction Concept}
\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Transaction Concept\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Transaction Concept}
    \begin{itemize}[<+->]
        \item A transaction is a unit of program execution that accesses and, possibly updates, various data items
        \item For example, transaction to transfer \$50 from account A to account B:
        \begin{enumerate}
            \item \texttt{read(A)}
            \item \texttt{A := A - 50}
            \item \texttt{write(A)}
            \item \texttt{read(B)}
            \item \texttt{B := B + 50}
            \item \texttt{write(B)}
        \end{enumerate}
        \item Two main issues to deal with:
        \begin{itemize}[<+->]
            \item Failures of various kinds, such as hardware failures and system crashes
            \item Concurrent execution of multiple transactions
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Required Properties of a Transaction: ACID: Atomicity}
    \textbf{Atomicity Requirement}
    \begin{itemize}[<+->]
        \item If the transaction fails after step 3 and before step 6, money will be 'lost' leading to an inconsistent database state
        \begin{itemize}[<+->]
            \item $\triangleright$ Failure could be due to software or hardware
        \end{itemize}
        \item The system should ensure that updates of a partially executed transaction are not reflected in the database
    \end{itemize}
\end{frame}

\begin{frame}{Required Properties of a Transaction: ACID: Consistency}
    \textbf{Consistency Requirement}
    \begin{itemize}[<+->]
        \item A + B must be unchanged by the execution of the transaction
        \item In general, consistency requirements include
        \begin{itemize}[<+->]
            \item $\triangleright$ Explicitly specified integrity constraints
            \begin{itemize}[<+->]
                \item -- primary keys and foreign keys
            \end{itemize}
            \item $\triangleright$ Implicit integrity constraints
            \begin{itemize}[<+->]
                \item -- sum of balances of all accounts, minus sum of loan amounts must equal value of cash-in-hand
            \end{itemize}
        \end{itemize}
        \item A transaction, when starting to execute, must see a consistent database
        \item During transaction execution the database may be temporarily inconsistent
        \item When the transaction completes successfully the database must be consistent
        \begin{itemize}[<+->]
            \item $\triangleright$ Erroneous transaction logic can lead to inconsistency
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Required Properties of a Transaction: ACID: Isolation}
    \textbf{Isolation Requirement}
    \begin{itemize}[<+->]
        \item If between steps 3 and 6 (of the fund transfer transaction), another transaction T2 is allowed to access the partially updated database, it will see an inconsistent database (the sum A + B will be less than it should be)
    \end{itemize}
    
    \begin{center}
    \begin{tabular}{cc}
        T1 & T2 \\
        \midrule
        1. \texttt{read(A)} & \\
        2. \texttt{A := A - 50} & \\
        3. \texttt{write(A)} & \\
         & \texttt{read(A)}, \texttt{read(B)}, \texttt{print(A + B)} \\
        4. \texttt{read(B)} & \\
        5. \texttt{B := B + 50} & \\
        6. \texttt{write(B)} & \\
    \end{tabular}
    \end{center}
    
    \begin{itemize}[<+->]
        \item Isolation can be ensured trivially by running transactions serially
        \begin{itemize}[<+->]
            \item $\triangleright$ That is, one after the other
        \end{itemize}
        \item However, executing multiple transactions concurrently has significant benefits
    \end{itemize}
\end{frame}

\begin{frame}{Required Properties of a Transaction: ACID: Durability}
    \textbf{Durability Requirement}
    \begin{itemize}[<+->]
        \item Transaction to transfer \$50 from account A to account B:
        \item Once the user has been notified that the transaction has completed (that is, the transfer of the \$50 has taken place), the updates to the database by the transaction must persist even if there are software or hardware failures
    \end{itemize}
\end{frame}

\begin{frame}{ACID Properties}
    A transaction is a unit of program execution that accesses and possibly updates various data items:
    \begin{itemize}[<+->]
        \item \textbf{Atomicity}: Atomicity guarantees that each transaction is treated as a single \textit{unit}, which either succeeds completely, or fails completely
        \begin{itemize}[<+->]
            \item If any of the statements constituting a transaction fails to complete, the entire transaction fails and the database is left unchanged
            \item Atomicity must be guaranteed in every situation, including power failures, errors and crashes
        \end{itemize}
        \item \textbf{Consistency}: Consistency ensures that a transaction can only bring the database from one valid state to another, maintaining database invariants
        \begin{itemize}[<+->]
            \item Any data written to the database must be valid according to all defined rules, including constraints, cascades, triggers, and any combination thereof
        \end{itemize}
        \item \textbf{Isolation}: Transactions are often executed concurrently (multiple transactions reading and writing to a table at the same time)
        \begin{itemize}[<+->]
            \item Isolation ensures that concurrent execution of transactions leaves the database in the same state that would have been obtained if the transactions were executed sequentially
        \end{itemize}
        \item \textbf{Durability}: Durability guarantees that once a transaction has been committed, it will remain committed even in the case of a system failure (like power outage or crash)
        \begin{itemize}[<+->]
            \item This usually means that completed transactions (or their effects) are recorded in non-volatile memory
        \end{itemize}
    \end{itemize}
\end{frame}

\section{Transaction States}
\begin{frame}
  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}Transaction States\par%
  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Transaction States}
    \begin{itemize}[<+->]
        \item Every transaction can be in one of the following states (like Process States in OS)
        \item \textbf{Active}
        \begin{itemize}[<+->]
            \item $\triangleright$ The initial state; the transaction stays in this state while it is executing
        \end{itemize}
        \item \textbf{Partially committed}
        \begin{itemize}[<+->]
            \item $\triangleright$ After the final statement has been executed
        \end{itemize}
        \item \textbf{Failed}
        \begin{itemize}[<+->]
            \item $\triangleright$ After the discovery that normal execution can no longer proceed
        \end{itemize}
        \item \textbf{Aborted}
        \begin{itemize}[<+->]
            \item $\triangleright$ After the transaction has been rolled back and the database restored to its state prior to the start of the transaction. Two options after it has been aborted:
            \item -- Restart the transaction: Can be done only if no internal logical error
            \item -- Kill the transaction
        \end{itemize}
        \item \textbf{Committed}
        \begin{itemize}[<+->]
            \item $\triangleright$ After successful completion
        \end{itemize}
        \item \textbf{Terminated}
        \begin{itemize}[<+->]
            \item $\triangleright$ After it has been committed or aborted (killed)
        \end{itemize}
    \end{itemize}
\end{frame}

  \vfill

\begin{frame}{Concurrent Executions}
    \begin{itemize}[<+->]
        \item Multiple transactions are allowed to run concurrently in the system. Advantages are:
        \item Increased processor and disk utilization, leading to better transaction throughput
        \begin{itemize}[<+->]
            \item $\triangleright$ For example, one transaction can be using the CPU while another is reading from or writing to the disk
        \end{itemize}
        \item Reduced average response time for transactions: short transactions need not wait behind long ones
        \item \textbf{Concurrency Control Schemes}: Mechanisms to achieve isolation
        \item To control the interaction among the concurrent transactions in order to prevent them from destroying the consistency of the database
    \end{itemize}
\end{frame}

\begin{frame}{Schedules}
    \begin{itemize}[<+->]
        \item \textbf{Schedule}: A sequence of instructions that specify the chronological order in which instructions of concurrent transactions are executed
        \item A schedule for a set of transactions must consist of all instructions of those transactions
        \item Must preserve the order in which the instructions appear in each individual transaction
        \item A transaction that successfully completes its execution will have a \texttt{commit} instructions as the last statement
        \item By default transaction assumed to execute \texttt{commit} instruction as its last step
        \item A transaction that fails to successfully complete its execution will have an \texttt{abort} instruction as the last statement
    \end{itemize}
\end{frame}

\begin{frame}{Module Summary}
    \begin{block}{}
        \begin{itemize}[<+->]

            \item Transactions are executed in concurrent fashion for better throughput
            \item Concurrent execution of transactions raise serializability issues that need to be addressed
            \item All schedules may not satisfy ACID properties
