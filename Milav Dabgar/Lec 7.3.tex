\begin{frame}{Working with SQL and Native Language}

        \begin{itemize}
            \item[$\triangleright$] Connect with the database server
            \item[$\triangleright$] Send SQL commands to the database server
            \item[$\triangleright$] Fetch tuples of result one-by-one into program variables

        \item \textbf{Frameworks}
        \begin{itemize}
            \item[$\triangleright$] \textbf{Connectionist:} ODBC (C, C++, C\#, Visual Basic, Python), JDBC (Java)
            \item[$\triangleright$] \textbf{Embedding:} Embedded SQL works with C, C++, Java, COBOL, FORTRAN and Pascal
        \end{itemize}
    \end{itemize}

\begin{frame}{ODBC}
    \begin{itemize}[<+->]
        \item Open Database Connectivity (ODBC) is a standard API for accessing DBMS
        \item It aimed to be independent of database systems and operating systems
        \item An application written using ODBC can be ported to other platforms with few changes to the data access code
        \item ODBC is
        \begin{itemize}
            \item[$\triangleright$] A standard for application program to communicate with a database server
            \item[$\triangleright$] An API to Open a connection, Send queries and updates, Get back results
        \end{itemize}
        \item Applications such as GUI, Spreadsheets, etc. can use ODBC
    \end{itemize}
\end{frame}

\begin{frame}{JDBC}
    \begin{itemize}[<+->]
        \item Java Database Connectivity (JDBC) is an API for Java which defines how a client may access a database
        \item JDBC supports a variety of features for querying and updating data, and for retrieving query results; metadata retrieval
        \item Model for communicating with the database:
        \begin{itemize}
            \item[$\triangleright$] Open a connection
            \item[$\triangleright$] Create a 'statement' object
            \item[$\triangleright$] Execute queries using the Statement object to send queries and fetch results
            \item[$\triangleright$] Exception mechanism to handle errors
        \end{itemize}
    \end{itemize}
\end{frame}

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;

public class RetrieveResultSet {
  public static void main(String[] args) {
    String connectionUrl = "jdbc:sqlserver://<server>:<port>;databaseName=AdventureWorks;";
    // Create a variable for the connection string.
    connectionUrl += "user=<user>; password=<password>";
    try (Connection con = DriverManager.getConnection(connectionUrl);
         Statement stmt = con.createStatement();) {
      createTable(stmt);
      String SQL = "SELECT * FROM Production.Product;";
      ResultSet rs = stmt.executeQuery(SQL);
      displayRow("PRODUCTS", rs);
    }
    // Handle any errors that may have occurred.
    catch (SQLException e) {
      e.printStackTrace();
    }
  }
}

\begin{frame}{Connectionist Bridge Configurations}

        \item A bridge is a special kind of driver that uses another driver-based technology; translates source function-calls into target function-calls
        \item Common bridges are:
        \begin{itemize}
            \item[$\triangleright$] \textbf{ODBC-to-JDBC bridges}: ODBC driver using JDBC services (e.g., OpenLink ODBC-JDBC Bridge)

            \item[$\triangleright$] \textbf{OLE DB-to-ODBC bridges}: OLE DB Provider using ODBC services
            \item[$\triangleright$] \textbf{ADO.NET-to-ODBC bridges}: ADO.NET Provider using ODBC services

\begin{frame}{Embedded SQL}
    \begin{itemize}
        \item The SQL standard defines embedding of SQL in a variety of programming languages such as C, C++, Java, FORTRAN, and PL/1
        \item A language to which SQL queries are embedded is referred to as a \textbf{host language}, and the SQL structures permitted in the host language comprise \textbf{embedded SQL}
        \item The basic form of these languages follows that of the System R embedding of SQL into PL/1
        \item \texttt{EXEC SQL} (or similar alternate like \texttt{\#sql}) statement is used to identify embedded SQL request to the pre-processor
        \item \texttt{EXEC SQL < embedded SQL statement > ;}
        \item \textbf{Note:} this varies by language:
        \begin{itemize}
            \item In some languages, like COBOL, the semicolon is replaced with \texttt{END-EXEC}
            \item In Java embedding uses \texttt{\# SQL \{ .... \} ;}
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Embedded SQL (2)}
    \begin{itemize}
        \item Before executing any SQL statements, the program must first connect to the database. This is done using:
        \begin{verbatim}
EXEC-SQL connect to server user user-name using password;
        \end{verbatim}
        Here, \texttt{server} identifies the server to which a connection is to be established
        \item Variables of the host language can be used within embedded SQL statements. They are preceded by a colon (\texttt{:}) to distinguish from SQL variables (for example, \texttt{:credit\_amount})
        \item Variables used as above must be declared within \texttt{DECLARE} section, as illustrated below. The syntax for declaring the variables, however, follows the usual host language syntax
        \begin{verbatim}
EXEC-SQL BEGIN DECLARE SECTION
    int credit_amount;
EXEC-SQL END DECLARE SECTION;
        \end{verbatim}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Embedded SQL (3)}
    \begin{itemize}
        \item To write an embedded SQL query, we use the \texttt{declare c cursor for < SQL query >} statement. The variable \texttt{c} is used to identify the query
        \item \textbf{Example:}
        \begin{itemize}
            \item From within a host language, find the ID and name of students who have completed more than the number of credits stored in variable \texttt{credit\_amount} in the host language
            \item Specify the query in SQL as follows:
            \begin{verbatim}
EXEC SQL
    declare c cursor for
    select ID, name
    from student
    where tot_cred > :credit_amount
END EXEC
            \end{verbatim}
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Embedded SQL (4)}
    \begin{itemize}
        \item \textbf{Example}
        \begin{itemize}
            \item From within a host language, find the ID and name of students who have completed more than the number of credits stored in variable \texttt{credit\_amount} in the host language
            \item Specify the query in SQL as follows:
            \begin{verbatim}
EXEC SQL declare c cursor for
    select ID, name from student

            \end{verbatim}
        \end{itemize}
        \item The variable \texttt{c} (used in the cursor declaration) is used to identify the query
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Embedded SQL (5)}
    \begin{itemize}
        \item The \texttt{open} statement for our example is as follows:
        \begin{verbatim}
EXEC SQL open c ;
        \end{verbatim}
        This statement causes the database system to execute the query and to save the results within a temporary relation. The query uses the value of the host-language variable \texttt{credit-amount} at the time the \texttt{open} statement is executed.
        \item The \texttt{fetch} statement causes the values of one tuple in the query result to be placed on host language variables.
        \begin{verbatim}
EXEC SQL fetch c into :si, :sn END EXEC
        \end{verbatim}
        Repeated calls to \texttt{fetch} get successive tuples in the query result
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Embedded SQL (6)}
    \begin{itemize}
        \item A variable called \texttt{SQLSTATE} in the SQL communication area (\texttt{SQLCA}) gets set to \texttt{'02000'} to indicate no more data is available
        \item The \texttt{close} statement causes the database system to delete the temporary relation that holds the result of the query.
        \begin{verbatim}
EXEC SQL close c ;
        \end{verbatim}
        \textbf{Note:} above details vary with language. For example, the Java embedding defines Java iterators to step through result tuples.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Embedded SQL (7): Updates}
    \begin{itemize}
        \item Embedded SQL expressions for database modification (\texttt{update}, \texttt{insert}, and \texttt{delete})
        \item Can update tuples fetched by cursor by declaring that the cursor is for update
        \begin{verbatim}
EXEC SQL
  declare c cursor for
  select * from instructor
  where dept_name = 'Music'
  for update
        \end{verbatim}
        \item We then iterate through the tuples by performing \texttt{fetch} operations on the cursor (as illustrated earlier), and after fetching each tuple we execute the following code:
        \begin{verbatim}
update instructor
set salary = salary + 1000
where current of c
        \end{verbatim}
    \end{itemize}
\end{frame}

#include <stdio.h>

#define EXIT 0
#define NOEXIT 1

EXEC SQL INCLUDE SQLCA; // Include DB2's SQL error reporting facility.

EXEC SQL BEGIN DECLARE SECTION; // Declare the SQL interface variables.
  short sage, sid;
  char sname[16];
EXEC SQL END DECLARE SECTION;

// Declare variables to be used in the following C program
char msg[1025];
int rc, errcount;

// This macro prints the message in the SQLCA if the return code is 0 and the SQLCODE is not 0
#define PRINT_MESSAGE() { \
  if (rc == 0 && sqlca.sqlcode != 0) { \
    sqlaintp(msg, 1024, 0, &sqlca); \
    printf("%s\n",msg); \
  } \
}

\begin{frame}[fragile]{Embedded SQL: C Example (3)}

// The macro prints out all fields in the SQLCA
#define DUMP_SQLCA() { \
  printf("********** DUMP OF SQLCA **********************\n");
  ...
}

// This macro prints the message in the SQLCA if one exists
// If the return code is not 0 or the SQLCODE is not expected, an error occurred and must be recorded.
#define CHECK_SQL(code,text_string,eExit) { \
  PRINT_MESSAGE(); \
  if (rc != 0 || sqlca.sqlcode != code) { \
    printf("%s\n",text_string); \
    printf("Expected code = %d\n",code); \
    if (rc == 0) DUMP_SQLCA(); \
    else printf("RC: %d\n",rc); \
    errcount += 1; \

}

\begin{frame}{Embedded SQL: Java Example (5): Notes}
    \begin{itemize}
        \item \textbf{2 Initialize the iterator}. The iterator object \texttt{cursor1} is initialized using the result of a query. The query stores the result in \texttt{cursor1}.
        \item \textbf{3 Advance the iterator to the next row}. The \texttt{cursor1.next()} method returns a Boolean \texttt{false} if there are no more rows to retrieve.
        \item \textbf{4 Move the data}. The named accessor method \texttt{empno()} returns the value of the column named \texttt{empno} on the current row. The named accessor method \texttt{firstnme()} returns the value of the column named \texttt{firstnme} on the current row.
        \item \textbf{5 SELECT data into a host variable}. The \texttt{SELECT} statement passes the number of rows in the table into the host variable \texttt{count1}.
        \item \textbf{9 Close the iterators}. The \texttt{close()} method releases any resources held by the iterators. You should explicitly close iterators to ensure that system resources are released in a timely fashion.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Embedded SQL: Java Example (6)}
    \begin{verbatim}
// retrieve the updated data from the database
System.out.println("Retrieve the updated data from the database.");
str1 = "000010";
#sql cursor2 = {SELECT firstnme FROM employee WHERE empno = :str1}; // 6
// display the result set. cursor2.next() returns false when there are no more rows
System.out.println("Received results:");
while (true) {
  #sql { FETCH :cursor2 INTO :str2 }; // 7
  if (cursor2.endFetch()) break; // 8
  System.out.println(" empno= " + str1 + " firstname= " + str2 + "");
}
cursor2.close(); // 9

// rollback the update
System.out.println("Rollback the update.");
#sql { ROLLBACK work };
System.out.println("Rollback done.");
} // try
catch( Exception e ) {
  e.printStackTrace();
}
} // main
} // class App
    \end{verbatim}
\end{frame}

\begin{frame}{Embedded SQL: Java Example (7): Notes}
    \begin{itemize}
        \item \textbf{6 Initialize the iterator}. The iterator object \texttt{cursor2} is initialized using the result of a query. The query stores the result in \texttt{cursor2}.
        \item \textbf{7 Retrieve the data}. The \texttt{FETCH} statement returns the current value of the first column declared in the ByPos cursor from the result table into the host variable \texttt{str2}.
        \item \textbf{8 Check the success of a FETCH...INTO statement}. The \texttt{endFetch()} method returns a Boolean \texttt{true} if the iterator is not positioned on a row, that is, if the last attempt to fetch a row failed. The \texttt{endFetch()} method returns \texttt{false} if the last attempt to fetch a row was successful. DB2 attempts to fetch a row when the \texttt{next()} method is called. A \texttt{FETCH...INTO} statement implicitly calls the \texttt{next()} method.
        \item \textbf{9 Close the iterators}. The \texttt{close()} method releases any resources held by the iterators. You should explicitly close iterators to ensure that system resources are released in a timely fashion.
    \end{itemize}
