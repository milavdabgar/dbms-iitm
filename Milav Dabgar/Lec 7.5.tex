\section{Rapid Application Development}
\begin{frame}
  \vfill

\begin{frame}{Rapid Application Development}
    \begin{itemize}[<+->]
        \item A lot of effort is required to develop Web application interfaces, especially the rich interaction functionality associated with Web 2.0 applications
        \item Several approaches to speed up application development
        \begin{itemize}
            \item[$\triangleright$] Function library to generate user-interface elements
            \item[$\triangleright$] Drag-and-drop features in an IDE to create user-interface elements
            \item[$\triangleright$] Automatically generate code for user interface from a declarative specification
        \end{itemize}
        \item RAD Software is an agile model that focuses on fast prototyping and quick feedback
        \item App development has 4 phases: business modeling, data modeling, process modeling, and testing \& turnover
        \item With RAD, the time between prototypes and iterations is short, and integration occurs since inception
    \end{itemize}
\end{frame}

\begin{frame}{Rapid Application Development (2)}
    \begin{itemize}[<+->]
        \item \textbf{Web application development frameworks}
        \begin{itemize}
            \item[$\triangleright$] \textbf{Java Server Faces (JSF):} APIs for representing UI components, managing state, handling events, input validation, and page navigation
            \item[$\triangleright$] \textbf{Ruby on Rails:} Easy creation of CRUD interfaces by code generation from database schema or object model
        \end{itemize}
        \item \textbf{RAD Platforms and Tools}
        \begin{itemize}
            \item[$\triangleright$] G Suite, Google App Engine
            \item[$\triangleright$] Microsoft Azure, Amazon EC2, AWS Elastic Beanstalk
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{ASP.NET and Visual Studio}
    \begin{itemize}[<+->]
        \item ASP.NET provides a variety of controls that are interpreted at server, and generate HTML code
        \item Visual Studio provides drag-and-drop development using these controls
        \begin{itemize}
            \item[$\triangleright$] For example, menus and list boxes can be associated with \texttt{DataSet} object
        \end{itemize}
        \item Validator controls (constraints) can be added to form input fields
        \begin{itemize}
            \item[$\triangleright$] JavaScript to enforce constraints at client, and separately enforced at server
        \end{itemize}
        \item User actions such as selecting a value from a menu can be associated with actions at server
        \item DataGrid provides convenient way of displaying SQL query results in tabular format
    \end{itemize}
\end{frame}

\begin{frame}

  \end{beamercolorbox}
  \vfill
\end{frame}

\begin{frame}{Application Performance}
    \begin{itemize}[<+->]
        \item Performance is an issue for popular Web sites
        \begin{itemize}
            \item[$\triangleright$] May be accessed by millions of users every day, thousands of requests per second at peak time
        \end{itemize}
        \item Caching techniques used to reduce cost of serving pages by exploiting commonalities between requests
        \item At the server site:
        \begin{itemize}
            \item[$\triangleright$] Caching of JDBC connections between servlet requests (connection pooling)
            \item[$\triangleright$] Caching results of database queries (must be updated if underlying database changes)
            \item[$\triangleright$] Caching of generated HTML
        \end{itemize}
        \item At the client's network
        \begin{itemize}
            \item[$\triangleright$] Caching of pages by Web proxy
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Application Security: SQL Injection}
    \begin{itemize}
        \item Suppose query is constructed using
        \begin{center}\texttt{"select * from instructor where name = '" + name + "'"}\end{center}
        \item Suppose the user, instead of entering a name, enters:
        \begin{center}\texttt{X' or 'Y' = 'Y}\end{center}
        \item then the resulting statement becomes:
        \begin{center}\texttt{"select * from instructor where name = '" + "X' or 'Y' = 'Y" + "'"}\end{center}
        \item which is:
        \begin{itemize}
            \item $\triangleright$ \texttt{select * from instructor where name = 'X' or 'Y' = 'Y'}
        \end{itemize}
        \item User could have even used
        \begin{itemize}
            \item $\triangleright$ \texttt{X'; update instructor set salary = salary + 10000; -{}-}
        \end{itemize}
        \item Prepared statement internally uses:
        \begin{center}\texttt{select * from instructor where name = 'X \textbackslash' or \textbackslash'Y \textbackslash' = \textbackslash'Y'}\end{center}
        \item \textbf{Always use prepared statements, with user inputs as parameters}
        \item Is the following prepared statement secure?
        \begin{center}\texttt{conn.prepareStatement("select * from instructor where name = '" + name + "'")}\end{center}
    \end{itemize}
\end{frame}

\begin{frame}{Application Security (2): Password Leakage}
    \begin{alertblock}{Warning}
        Never store passwords in clear text in scripts that may be accessible to users
    \end{alertblock}
    \begin{itemize}[<+->]
        \item For example, in files in a directory accessible to a web server
        \begin{itemize}
            \item[$\triangleright$] Web server will execute but not provide source of script files such as \texttt{file.jsp}, but source of editor backup files such as \texttt{file.jsp\~{}} may be served
        \end{itemize}
        \item Restrict access to database server from IPs of machines running application servers
        \begin{itemize}
            \item[$\triangleright$] Most databases allow restriction of access by source IP address
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Application Security (3): Authentication}
    \begin{itemize}[<+->]
        \item Single factor authentication such as passwords too risky for critical applications
        \begin{itemize}
            \item[$\triangleright$] Vulnerable to: guessing, packet sniffing, password reuse across sites, spyware
        \end{itemize}
        \item \textbf{Two-factor authentication}
        \begin{itemize}
            \item[$\triangleright$] Password plus one-time password sent by SMS
            \item[$\triangleright$] Password plus hardware one-time password device
            \begin{itemize}
                \item device generates a new pseudo-random number every minute
                \item application server generates same sequence to verify
            \end{itemize}
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Application Security (4): Application-Level Authorization}
    \begin{itemize}
        \item Current SQL standard does not allow fine-grained authorization such as 'students can see their own grades, but not other's grades'
        \item \textbf{Problem 1:} Database has no idea who are application users
        \item \textbf{Problem 2:} SQL authorization is at the level of tables, or columns of tables, but not to specific rows of a table
        \item One workaround: use views such as
        \begin{verbatim}
create view studentTakes as
select * from takes
where takes.ID = syscontext.user_id()
        \end{verbatim}
        \item where \texttt{syscontext.user\_id()} provides end user identity
        \begin{itemize}
            \item $\triangleright$ end user identity must be provided to the database by the application
        \end{itemize}
        \item Having multiple such views is cumbersome
    \end{itemize}
\end{frame}

\begin{frame}{Application Security (5): Application-Level Authorization}
    \begin{itemize}[<+->]
        \item Currently, authorization is done entirely in application
        \begin{itemize}
            \item[$\triangleright$] Entire application code has access to entire database -- large surface area, making protection harder
        \end{itemize}
        \item Alternative: fine-grained (row-level) authorization schemes
        \begin{itemize}
            \item[$\triangleright$] Extensions to SQL authorization proposed but not currently implemented
            \item[$\triangleright$] Oracle Virtual Private Database (VPD) allows predicates to be added transparently to all SQL queries
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Application Security (6): Audit Trails}
    \begin{itemize}[<+->]
        \item Applications must log actions to an audit trail, to detect who carried out an update, or accessed some sensitive data
        \item Audit trails used after-the-fact to
        \begin{itemize}
            \item[$\triangleright$] detect security breaches
            \item[$\triangleright$] repair damage caused by security breach
            \item[$\triangleright$] trace who carried out the breach
        \end{itemize}
        \item Audit trails needed at
        \begin{itemize}
            \item[$\triangleright$] Database level, and at
            \item[$\triangleright$] Application level
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{What is a Mobile App?}
    \begin{itemize}[<+->]
        \item A type of application software designed to run on a mobile device, such as a smartphone or tablet computer
        \item Designed with consideration for device demands and constraints, and to take advantage of specialized capabilities
        \item \textbf{Constraints:}
        \begin{itemize}
            \item[$\triangleright$] Form Factor -- influences display and navigation
            \item[$\triangleright$] Limited Memory, Computing Power, and Battery
            \item[$\triangleright$] Limited Bandwidth
        \end{itemize}
        \item \textbf{Advantages:}
        \begin{itemize}
            \item[$\triangleright$] Availability of sensors like accelerometer
            \item[$\triangleright$] Touchscreen -- Gesture-based Navigation
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Mobile Website vis-\`a-vis Mobile App}
    \begin{columns}[T]
        \column{0.5\textwidth}
        \begin{block}{Mobile Website}
            \begin{itemize}
                \item Browser-based HTML pages
                \item Accessed over WiFi/3G/4G
                \item Designed for small touch-screen
                \item Can use click-to-call and location mapping
            \end{itemize}
        \end{block}
        \column{0.5\textwidth}
        \begin{exampleblock}{Mobile Apps}
            \begin{itemize}
                \item Downloaded and installed on device
                \item From App Store, Google Play, etc.
                \item Can pull content from Internet or work offline
            \end{itemize}
        \end{exampleblock}
    \end{columns}
\end{frame}

\begin{frame}{Architecture of Mobile App}
    \begin{itemize}[<+->]
        \item Typically \textbf{3-tier architecture}
        \begin{itemize}
            \item[$\triangleright$] Presentation, Business, Data layers
        \end{itemize}
        \item Data Layer is often split between:
        \begin{itemize}
            \item[$\triangleright$] Local Data (on-device storage)
            \item[$\triangleright$] Remote Data (server/cloud)
        \end{itemize}
        \item Needs customization for platform: Android, iOS, Windows
    \end{itemize}
\end{frame}

\begin{frame}{Types of Mobile Apps}
    \begin{itemize}[<+->]
        \item \textbf{Native Apps}: Completely written in the native language of a platform
        \begin{itemize}
            \item[$\triangleright$] iOS $\rightarrow$ Objective-C; Android $\rightarrow$ Java or C/C++
            \item[$\triangleright$] Platform specific (heavily dependent on OS)
        \end{itemize}
        \item \textbf{Web Apps}: Run completely inside of a Web browser
        \begin{itemize}
            \item[$\triangleright$] Interfaces built with HTML/CSS, powered by Ruby on Rails, JavaScript, PHP, or Python
            \item[$\triangleright$] Portable across any phone, tablet, or computer
        \end{itemize}
        \item \textbf{Hybrid Apps}: Combines attributes of both native and Web apps
        \begin{itemize}
            \item[$\triangleright$] Uses common cross-platform code while tailoring native-specific attributes
        \end{itemize}
    \end{itemize}

\end{frame}
