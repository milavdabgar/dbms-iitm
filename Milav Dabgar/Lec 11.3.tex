\begin{frame}{Hot Backup: Recap}
    \begin{itemize}[<+->]
        \item In systems where high availability is a requirement Hot backup is preferable wherever possible
        \item Hot backup refers to keeping a database up and running while the backup is performed concurrently
        \item Such a system usually has a module or plug-in that allows the database to be backed up while staying available to end users
        \item Databases which stores transactions of asset management companies, hedge funds, high frequency trading companies etc. try to implement Hot backups as these data are highly dynamic and the operations run 24x7
        \item Real time systems like sensor and actuator data in embedded devices, satellite transmissions etc. also use Hot backup
    \end{itemize}
\end{frame}

\begin{frame}{Transactional Logging as Hot Backup}
    \begin{itemize}[<+->]
        \item In regular database systems, Hot Backup is mainly used for Transaction Log Backup
        \item Cold backup strategies like Differential, Incremental are preferred for Data backup. The reason is evident from the disadvantages of Hot backup
        \item Transactional Logging is used in circumstances where a possibly inconsistent backup is taken, but another file generated and backed up (after the database file has been fully backed up) can be used to restore consistency
        \item The information regarding data backup versions while recovery at a given point can be inferred from the Transactional Log backup set
        \item Thus they play a vital role in database recovery
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Schemes}
    \begin{itemize}[<+->]
        \item \textbf{So far:}
        \begin{itemize}
            \item[$\triangleright$] We covered key concepts
            \item[$\triangleright$] We assumed serial execution of transactions
        \end{itemize}
        \item \textbf{Now:}
        \begin{itemize}
            \item[$\triangleright$] We discuss concurrency control issues
            \item[$\triangleright$] We present the components of the basic recovery algorithm
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Concurrency Control and Recovery}
    \begin{itemize}[<+->]
        \item With concurrent transactions, all transactions share a single disk buffer and a single log
        \item A buffer block can have data items updated by one or more transactions
        \item We assume that if a transaction $T_i$ has modified an item, no other transaction can modify the same item until $T_i$ has committed or aborted
        \item That is, the updates of uncommitted transactions should not be visible to other transactions
        \begin{itemize}
            \item[$\triangleright$] Otherwise how do we perform undo if $T_1$ updates A, then $T_2$ updates A and commits, and finally $T_1$ has to abort?
        \end{itemize}
        \item Can be ensured by obtaining exclusive locks on updated items and holding the locks till end of transaction (strict two-phase locking)
        \item Log records of different transactions may be interspersed in the log
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithm}
    \begin{itemize}[<+->]
        \item \textbf{Logging (during normal operation):}
        \begin{itemize}
            \item[$\triangleright$] \texttt{<$T_i$ start>} at transaction start
            \item[$\triangleright$] \texttt{<$T_i$, $X_j$, $V_1$, $V_2$>} for each update, and
            \item[$\triangleright$] \texttt{<$T_i$ commit>} at transaction end
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithm (2)}
    \begin{itemize}[<+->]
        \item \textbf{Transaction rollback (during normal operation)}
        \item Let $T_i$ be the transaction to be rolled back
        \item Scan log backwards from the end, and for each log record of $T_i$ of the form \texttt{<$T_i$, $X_j$, $V_1$, $V_2$>}
        \begin{itemize}
            \item[$\triangleright$] perform the undo by writing $V_1$ to $X_j$,
            \item[$\triangleright$] write a log record \texttt{<$T_i$, $X_j$, $V_1$>}
            \item[$\triangleright$] \dots such log records are called \textbf{Compensation Log Records (CLR)}
        \end{itemize}
        \item Once the record \texttt{<$T_i$ start>} is found stop the scan and write the log record \texttt{<$T_i$ abort>}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithm (3): Checkpoints Recap}
    \begin{itemize}[<+->]
        \item Let the time of checkpointing is $t_{\text{check}}$ and the time of system crash is $t_{\text{fail}}$
        \item Let there be four transactions $T_a$, $T_b$, $T_c$ and $T_d$ such that:
        \begin{itemize}
            \item[$\triangleright$] $T_a$ commits before checkpoint
            \item[$\triangleright$] $T_b$ starts before checkpoint and commits before system crash
            \item[$\triangleright$] $T_c$ starts after checkpoint and commits before system crash
            \item[$\triangleright$] $T_d$ starts after checkpoint and was active at the time of system crash
        \end{itemize}
        \item The actions that are taken by the recovery manager are:
        \begin{itemize}
            \item[$\triangleright$] Nothing is done with $T_a$
            \item[$\triangleright$] Transaction \textbf{redo} is performed for $T_b$ and $T_c$
            \item[$\triangleright$] Transaction \textbf{undo} is performed for $T_d$
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithm (4): Checkpoints Recap}
    \uncover<1->{
        \begin{center}
            \includegraphics[width=\textwidth]{"../Lecture 11.3_artifacts/image_000026_850cb78a5ce6be86f7d420e2e613258afc3818050a05c18f0fc0b7fec2cf7348.png"}
        \end{center}
    }
    \begin{itemize}[<+->]
        \item[<2->] Any transactions that committed before the last checkpoint should be ignored
        \begin{itemize}
            \item[$\triangleright$] $T_1$ can be ignored (updates already output to disk due to checkpoint)
        \end{itemize}
        \item[<3->] Any transactions that committed since the last checkpoint need to be redone
        \begin{itemize}
            \item[$\triangleright$] $T_2$ and $T_3$ redone
        \end{itemize}
        \item[<4->] Any transaction that was running at the time of failure needs to be undone and restarted
        \begin{itemize}
            \item[$\triangleright$] $T_4$ undone
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithm (5): Redo-Undo Phases}
    \begin{itemize}[<+->]
        \item \textbf{Recovery from failure:} Two phases
        \begin{itemize}
            \item[$\triangleright$] \textbf{Redo phase:} Replay updates of all transactions, whether they committed, aborted, or are incomplete
            \item[$\triangleright$] \textbf{Undo phase:} Undo all incomplete transactions
        \end{itemize}
    \end{itemize}
    \uncover<3->{
        \begin{center}
            \includegraphics[height=0.4\textheight]{"../Lecture 11.3_artifacts/image_000028_7b19a8bd321fc5381db10a95ca52da4a84cba4b84859fe3a71063f4a573ef5bf.png"}
        \end{center}
    }
\end{frame}

\begin{frame}{Recovery Algorithm (5): Redo-Undo Phases (cont.)}
    \begin{itemize}[<+->]
        \item \textbf{Requirement:}
        \begin{itemize}
            \item[$\triangleright$] Transactions of type $T_1$ need no recovery
            \item[$\triangleright$] Transactions of type $T_2$ or $T_4$ need to be redone
            \item[$\triangleright$] Transactions of type $T_3$ or $T_5$ need to be undone and restarted
        \end{itemize}
        \item \textbf{Strategy:}
        \begin{itemize}
            \item[$\triangleright$] Ignore $T_1$
            \item[$\triangleright$] Redo $T_2$, $T_3$, $T_4$ and $T_5$
            \item[$\triangleright$] Undo $T_3$ and $T_5$
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithm (6): Redo Phase}
    \begin{itemize}[<+->]
        \item Find last \texttt{<checkpoint L>} record, and set \texttt{undo-list} to L
        \item Scan forward from above \texttt{<checkpoint L>} record
        \begin{itemize}
            \item[$\triangleright$] Whenever a record \texttt{<$T_i$, $X_j$, $V_1$, $V_2$>} is found, redo it by writing $V_2$ to $X_j$
            \item[$\triangleright$] Whenever a log record \texttt{<$T_i$ start>} is found, add $T_i$ to \texttt{undo-list}
            \item[$\triangleright$] Whenever a log record \texttt{<$T_i$ commit>} or \texttt{<$T_i$ abort>} is found, remove $T_i$ from \texttt{undo-list}
        \end{itemize}
        \item Steps for the \textbf{REDO} operation are:
        \begin{itemize}
            \item[$\triangleright$] If the transaction has done INSERT, the recovery manager generates an insert from the log
            \item[$\triangleright$] If the transaction has done DELETE, the recovery manager generates a delete from the log
            \item[$\triangleright$] If the transaction has done UPDATE, the recovery manager generates an update from the log.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Recovery Algorithm (7): Undo Phase}
    \begin{itemize}[<+->]
        \item Scan log backwards from end
        \begin{itemize}
            \item[$\triangleright$] Whenever a log record \texttt{<$T_i$, $X_j$, $V_1$, $V_2$>} is found where $T_i$ is in \texttt{undo-list} perform same actions as for transaction rollback:
            \item[$\bullet$] Perform undo by writing $V_1$ to $X_j$
            \item[$\bullet$] Write a log record \texttt{<$T_i$, $X_j$, $V_1$>}
            \item[$\triangleright$] Whenever a log record \texttt{<$T_i$ start>} is found where $T_i$ is in \texttt{undo-list}
            \item[$\bullet$] Write a log record \texttt{<$T_i$ abort>}
            \item[$\bullet$] Remove $T_i$ from \texttt{undo-list}
        \end{itemize}
        \item Stop when \texttt{undo-list} is empty
        \begin{itemize}
            \item[$\triangleright$] That is, \texttt{<$T_i$ start>} has been found for every transaction in \texttt{undo-list}
        \end{itemize}
        \item Steps for the \textbf{UNDO} operation are:
        \begin{itemize}
            \item[$\triangleright$] If the faulty transaction has done INSERT, the recovery manager deletes the data item(s) inserted
            \item[$\triangleright$] If the faulty transaction has done DELETE, the recovery manager inserts the deleted data item(s) from the log
            \item[$\triangleright$] If the faulty transaction has done UPDATE, the recovery manager eliminates the value by writing the before-update value from the log
        \end{itemize}
        \item After undo phase completes, normal transaction processing can commence
    \end{itemize}
\end{frame}
