\begin{frame}{Data Structure}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Data structure}: A data structure specifies the way of organizing and storing in-memory data that enables efficient access and modification of the data.
            \item Linear Data Structures
            \item Non-linear Data Structures
            \item Most data structure has a container for the data and typical operations that its needs to perform
            \item For applications relating to data management, the key operations are:
            \begin{itemize}[<+->]
                \item Create
                \item Insert
                \item Delete
                \item Find / Search
                \item Close
            \end{itemize}
            \item Efficiency is measured in terms of time and space taken for these operations
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Linear Data Structures}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item A Linear data structure has data elements arranged in linear or sequential manner such that each member element is connected to its previous and next element.
            \item Since data elements are sequentially connected, each element is traversable through a single run.
            \item Examples of linear data structures are Array, Linked List, Queue, Stack, etc.
        \end{itemize}
    \end{block}

\begin{frame}{Linear Data Structures (2)}
    \begin{block}{Different examples of linear data structure:}
        \begin{itemize}[<+->]
            \item \textbf{Array} : The data elements are stored at contiguous locations in memory.
            \item \textbf{Linked List} : The data elements are not required to be stored at contiguous locations in memory. Rather each element stores a link (a pointer to a reference) to the location of the next element.
            \item \textbf{Queue} : It is a FIFO (First In First Out) data structure. The element that has been inserted first in the queue would be removed first. Thus, insert and removal of the elements in this take place in the same order.
            \item \textbf{Stack} : It is a LIFO (Last In First Out) data structure. The element that has been inserted last in the stack would be removed first. Thus, insert and removal of the elements in this take place in the reverse order.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Linear Data Structures (3): Array}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item The elements are stored in contiguous memory locations.
            \item Simple access using indices. For example, let the array name be arr, we can access the element at position 5 as arr[5].
            \item Array allows random access using its index which is fast (cost of $\mathcal{O}(1)$). Useful for operations like sorting, searching.
        \end{itemize}
    \end{block}

\begin{frame}{Linear Data Structures (4): Array}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Have fixed sizes, not flexible}. Since we do not know the number of elements to be stored in runtime, If we create it too large then it can be a waste of memory, if we create it too small then some elements may not be accommodated in the array.
            \item For example, suppose we create an array to store 8 elements. However, during execution of the program only 5 elements are available, which results in wastage of memory space.
        \end{itemize}

\begin{frame}{Linear Data Structures (7): Linked List}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item Elements are not required to be stored at contiguous memory locations. A new element can be stored anywhere in the memory where free space is available. Thus, it provides better memory usage than arrays.
            \item For each new element allocated, a link (a pointer or a reference) is created for the new element using which the element can be added to the linked list.
        \end{itemize}
    \end{block}
    \begin{center}
        \includegraphics[height=0.15\textheight, width=\textwidth, keepaspectratio]{"../Lecture 8.2_artifacts/image_000019_50e59e53b0febbd8701782d30f13f531db47e8c878f2d3d22f77795430836157.png"}
    \end{center}
    \begin{block}{}
        Each element is stored in a node. A node has two parts:
        \begin{itemize}[<+->]
            \item \textbf{Info} : stores the element.
            \item \textbf{Link} : stores the location of the next node.
            \item \textbf{Header} is a link to the first node of the linked list.
        \end{itemize}
    \end{block}

\begin{frame}{Linear Data Structures (8): Linked List}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item \textbf{Flexible in size}. Size of a linked list grows or shrinks as and when new elements are inserted or deleted.
            \item Random access is not possible in linked lists. The elements will have to be accessed sequentially.
            \item Insertion or removal of an element at/from any arbitrary position is efficient as none of the elements are not required to be moved to new locations.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Linear Data Structures (9): Linked List}
    \begin{block}{Insertion at front:}
        \begin{itemize}[<+->]
            \item Insertion or removal of an element at/from any arbitrary position is efficient.
            \item \textbf{Insertion at front:}
            \begin{enumerate}
                \item \texttt{NewNode.Link = Header}
                \item \texttt{Header = NewNode}
            \end{enumerate}
        \end{itemize}
    \end{block}

\begin{frame}{Linear Search}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item The algorithm starts with the first element, compares with the given key value and returns yes if they match.
            \item If it does not match, then it proceeds sequentially comparing each element of the list with the given key until a match is found or the full list is traversed.
        \end{itemize}
    \end{block}
    \begin{block}{Example}
        Let the given input list be \texttt{inputArr = ['a','c','a','d','e','m','i','c','s']} and the search key be \texttt{'i'}.
    \end{block}

\begin{frame}[fragile]{Linear Search (2)}
    \begin{exampleblock}{Python Code for Linear Search}
\begin{lstlisting}[language=Python]
def linSearch(inputArr, k): 
    for i in range(len(inputArr)): 
        if inputArr[i] == k: 
            return i 
    return -1

inputArr = ['a','c','a','d','e','m','i','c','s']
k = 'i'
index = linSearch(inputArr, k)
if index != -1:
    print("Element found at position " + str(index+1))
else:
    print("Not found")
\end{lstlisting}
    \end{exampleblock}
\end{frame}

\begin{frame}{Binary Search}
    \begin{block}{}
        \begin{itemize}[<+->]
            \item The input for the algorithm is a \textbf{sorted} list.
            \item The algorithm compares the key $k$ with the middle element in the list.
            \item If they match, then it returns the index.
            \item If the key does not match and is greater than the middle element, then the new list is the list to the right of the middle element.
            \item If the key does not match and is less than the middle element, then the new list is the list to the left of the middle element.
        \end{itemize}
    \end{block}
    \begin{block}{Example}
        Let the given input list be \texttt{inputArr = ['a','a','c','c','d','e','i','m','s']} and the search key be \texttt{'i'}.
    \end{block}

\end{frame}

\begin{frame}[fragile]{Binary Search (2)}
    \begin{exampleblock}{Python Code for Binary Search}
\begin{lstlisting}[language=Python]
def binSearch(inputArr, k): 
    low = 0 
    high = len(inputArr) - 1 
    while low <= high: 
        mid = (high + low) // 2  # Division(floor) 
        if inputArr[mid] < k:    # new list is to the right of k 
            low = mid + 1 
        elif inputArr[mid] > k:  # new list is to the left of k 
            high = mid - 1 
        else:                    # means k is present at mid 
            return mid 
    return -1                    # The element is not present

inputArr = ['a','a','c','c','d','e','i','m','s'] 
k = 'i' 
index = binSearch(inputArr, k) 
if index != -1: 
    print("Element found at position " + str(index+1)) 
else: 
    print("Not found ")
\end{lstlisting}
    \end{exampleblock}
\end{frame}
